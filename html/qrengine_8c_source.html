37//take in a QR byte array and output a formatted string with line breaks ready for console printing
38int outputQRStringFromQRBytes(const uint8_t* inQrBytes, char* outString)
39{
40 outString[0] = '\0'; //make sure it starts with a nul so strcat knows what to do
41 int size = qrcodegen_getSize(inQrBytes);
42 int border = 4;
43 const char* dark = " ";
44 const char* light = "##";
45
46 for (int y = -border; y < size + border; y++) {
47 for (int x = -border; x < size + border; x++) {
48 if (qrcodegen_getModule(inQrBytes, x, y)) {
49 strcat(outString, light);
50 } else {
51 strcat(outString, dark);
52 }
53 }
54 strcat(outString, "\n");
55 }
56 strcat(outString,"\n");
57 return 0;
58}

110uint8_t* bytesToRgb(uint8_t* qrBytes, size_t multiplier)
111{
112 size_t side = qrcodegen_getSize(qrBytes);
113 size_t pixelRunLength = side * side;
114 uint8_t darkPixel[3] = {0, 0, 0};
115 uint8_t litePixel[3] = {255, 255, 255};
116 int step = 3;
117
118 uint8_t* outRgb = (uint8_t*)calloc(pixelRunLength * step * multiplier * multiplier, sizeof(uint8_t));
119
120 int iterator = 0;
121
122 for (int y = 0; y < (int)side; y++) {
123 for (int r = 0; r < (int)multiplier; ++r) {
124 for (int x = 0; x < (int)side; x++) {
125 for (int n = 0; n < (int)multiplier; ++n) {
126 if (qrcodegen_getModule(qrBytes, x, y)) {
127 memcpy(outRgb + iterator, litePixel, step);
128 iterator = iterator + step;
129 } else {
130 memcpy(outRgb + iterator, darkPixel, step);
131 iterator = iterator + step;
132 }
133 }
134 }
135 }
136 }
137
138 // should have outRgb now... sized correctly
139 return outRgb;
140}

188

192 width = tempSideDim * sizeMultiplier;
193 height = tempSideDim * sizeMultiplier;
194
195 // bytes to rgb

197
198 unsigned error = lodepng_encode24(&png, &pngsize, image, width, height);
199 if (!error)

202 free(png);
203 free(image);
204 return (int)pngsize;
205 }
206 else
207 {
208 printf("png error %u: %s\n", error, lodepng_error_text(error));
209 free(png);
210 free(image);
211 return -1;
212 }
213 }
214 else
215 {
216 printf("Error generating QR\n");
217 return -1;
218 }
219}
