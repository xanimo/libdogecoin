94int main(int argc, char* argv[]) {
95 int ret = 0;
96 int long_index = 0;
97 int opt = 0;
98 char* data = 0;
99 char* ips = 0;
100 int debug = 0;
101 int timeout = 15;
102 int maxnodes = 10;
103 const dogecoin_chainparams* chain = &dogecoin_chainparams_main;
104
105 if (argc <= 1 || strlen(argv[argc - 1]) == 0 || argv[argc - 1][0] == '-') {
106 /* exit if no command was provided */
107 print_usage();
108 exit(EXIT_FAILURE);
109 }
110 data = argv[argc - 1];
111
112 /* get arguments */
113 while ((opt = getopt_long_only(argc, argv, "i:trds:m:", long_options, &long_index)) != -1) {
114 switch (opt) {
115 case 't':
116 chain = &dogecoin_chainparams_test;
117 break;
118 case 'r':
119 chain = &dogecoin_chainparams_regtest;
120 break;
121 case 'd':
122 debug = 1;
123 break;
124 case 's':
125 timeout = (int)strtol(optarg, (char**)NULL, 10);
126 break;
127 case 'i':
128 ips = optarg;
129 break;
130 case 'm':
131 maxnodes = (int)strtol(optarg, (char**)NULL, 10);
132 break;
133 case 'v':
134 print_version();
135 exit(EXIT_SUCCESS);
136 break;
137 default:
138 print_usage();
139 exit(EXIT_FAILURE);
140 }
141 }
142
143 /* The above code is checking if the data is NULL, empty or larger than the maximum
144 size of a p2p message. */
145 if (data == NULL || strlen(data) == 0 || strlen(data) > DOGECOIN_MAX_P2P_MSG_SIZE) {
146 return showError("Transaction in invalid or to large.\n");
147 }
148 uint8_t* data_bin = dogecoin_malloc(strlen(data) / 2 + 1);
149 size_t outlen = 0;
150 utils_hex_to_bin(data, data_bin, strlen(data), &outlen);
151
152 dogecoin_tx* tx = dogecoin_tx_new();
153 /* Deserializing the transaction and broadcasting it to the network. */
154 if (dogecoin_tx_deserialize(data_bin, outlen, tx, NULL)) {
155 broadcast_tx(chain, tx, ips, maxnodes, timeout, debug);
156 }
157 else {
158 showError("Transaction is invalid\n");
159 ret = 1;
160 }
161 dogecoin_free(data_bin);
162 dogecoin_tx_free(tx);
163
164 return ret;
165 }
