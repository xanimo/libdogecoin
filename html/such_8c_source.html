73void broadcasting_menu(int txindex, int is_testnet) {
74 int running = 1;
75 int selected = -1;
76 const dogecoin_chainparams* chain = is_testnet ? &dogecoin_chainparams_test : &dogecoin_chainparams_main;
77 working_transaction* tx = find_transaction(txindex);
78 char* raw_hexadecimal_tx = get_raw_transaction(txindex);
79 int length = HASH_COUNT(transactions);
80 while (running) {
81 printf("length: %d\n", length);
82 for (int i = 0; i <= length; i++) {
83 printf("\n--------------------------------\n");
84 printf("transaction to broadcast: %s\n", raw_hexadecimal_tx);
85 selected == i ? printf("confirm: [X]\n") : 0;
86
87 if (selected == i) {
88 printf("\n\n");
89 printf("please confirm this is the transaction you want to send:\n");
90 printf("1. yes\n");
91 printf("2. no\n");
92 switch (atoi(getl("\ncommand"))) {
93 case 1:
94 /* The above code is checking if the data is NULL, empty or larger than the maximum
95 size of a p2p message. */
96 if (raw_hexadecimal_tx == NULL || strlen(raw_hexadecimal_tx) == 0 || strlen(raw_hexadecimal_tx) > DOGECOIN_MAX_P2P_MSG_SIZE) {
97 printf("Transaction in invalid or to large.\n");
98 }
99 uint8_t* data_bin = dogecoin_malloc(strlen(raw_hexadecimal_tx) / 2 + 1);
100 size_t outlen = 0;
101 utils_hex_to_bin(raw_hexadecimal_tx, data_bin, strlen(raw_hexadecimal_tx), &outlen);
102
103 /* Deserializing the transaction and broadcasting it to the network. */
104 if (dogecoin_tx_deserialize(data_bin, outlen, tx->transaction, NULL)) {
105 broadcast_tx(chain, tx->transaction, 0, 10, 15, 0);
106 }
107 else {
108 printf("Transaction is invalid\n");
109 }
110 dogecoin_free(data_bin);
111 selected = -1; // set selected to number out of bounds for i
112 i = length; // reset loop to start
113 break;
114 case 2:
115 selected = -1; // set selected to number out of bounds for i
116 i = length; // reset loop to start
117 break;
118 }
119 }
120 // if on last iteration, jump into switch case pausing loop
121 // execution so user has ability to reset loop index in order
122 // to target desired input to edit. otherwise set loop index to
123 // length thus finishing final iteration and set running to 0 to
124 // escape encompassing while loop so we return to previous menu
125 if (i == length) {
126 printf("\n\n");
127 printf("1. broadcast transaction\n");
128 printf("2. main menu\n");
129 switch (atoi(getl("\ncommand"))) {
130 case 1:
131 // tx_input submenu
132
133 selected = i;
134 i = i - i - 1;
135 break;
136 case 2:
137 i = length;
138 running = 0;
139 break;
140 }
141 }
142 }
143 }
144 }

364void transaction_output_menu(int txindex, int is_testnet) {
365 int running_transaction_output_menu = 1;
366 while (running_transaction_output_menu) {
367 char* destinationaddress;
368 char* coin_amount[21];
369 dogecoin_mem_zero(coin_amount, 21);
370 uint64_t koinu_amount;
371 uint64_t tx_out_total = 0;
372 const dogecoin_chainparams* chain = is_testnet ? &dogecoin_chainparams_test : &dogecoin_chainparams_main;
373 working_transaction* tx = find_transaction(txindex);
374 int length = tx->transaction->vout->len;
375 int selected = -1;
376 printf("length: %d\n", length);
377 for (int i = 0; i < length; i++) {
378 dogecoin_tx_out* tx_out = vector_idx(tx->transaction->vout, i);
379 tx_out_total += tx_out->value;
380 printf("\n--------------------------------\n");
381 printf("output index: %d\n", i);
382 printf("script public key: %s\n", utils_uint8_to_hex((const uint8_t*)tx_out->script_pubkey->str, tx_out->script_pubkey->len));
383 koinu_to_coins_str(tx_out->value, (char*)coin_amount);
384 printf("amount: %s\n", (char*)coin_amount);
385 // selected should only equal anything other than -1 upon setting
386 // loop index in conditional targetting last iteration:
387 selected == i ? printf("selected: [X]\n") : 0;

389 printf("\n\n");
390 printf("1. select field to edit\n");
391 printf("2. finish editing\n");

393 case 1:
394 // tx_input submenu

396 printf("2. amount\n");

399 destinationaddress = (char*)getl("new destination address");
400 if (!verifyP2pkhAddress(destinationaddress, strlen(destinationaddress))) {
401 printf("\ninvalid destination address!\n");
402 break;
403 }
404 else {
405 koinu_amount = coins_to_koinu_str((char*)coin_amount);
406 vector_remove_idx(tx->transaction->vout, i);

414 tx_out->value = koinu_amount;
415 }
416 else {
417 printf("\namount is not a number!\n");
418 }
419 break;
420 }
421 tx_out_total = 0;
422 i = i - i - 1; // reset loop to start
423 break;
424 case 2:
425 selected = -1; // set selected to number out of bounds for i
426 tx_out_total = 0;
427 i = i - i - 1; // reset loop to start
428 break;
429 }
430 }
431 // if on last iteration, jump into switch case pausing loop
432 // execution so user has ability to reset loop index in order
433 // to target desired input to edit. otherwise set loop index to
434 // length thus finishing final iteration and set running to 0 to
435 // escape encompassing while loop so we return to previous menu
436 if (i == length - 1) {
437 printf("\n\n");
438 char* subtotal[21];
439 dogecoin_mem_zero(subtotal, 21);
440 koinu_to_coins_str(tx_out_total, (char*)subtotal);
441 printf("subtotal - desired fee: %s\n", (char*)subtotal);

443 printf("1. select output to edit\n");
444 printf("2. main menu\n");
