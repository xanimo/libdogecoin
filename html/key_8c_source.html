64dogecoin_bool dogecoin_privkey_gen(dogecoin_key* privkey)
65{
66 if (privkey == NULL)
67 return false;
68 do {
69 const dogecoin_bool res = dogecoin_random_bytes(privkey->privkey, DOGECOIN_ECKEY_PKEY_LENGTH, 0);
70 if (!res)
71 return false;
72 } while (dogecoin_ecc_verify_privatekey(privkey->privkey) == 0);
73 return true;
74}

76dogecoin_bool dogecoin_privkey_verify_pubkey(dogecoin_key* privkey, dogecoin_pubkey* pubkey)
77{
78 uint256 rnddata, hash;
79 const dogecoin_bool res = dogecoin_random_bytes(rnddata, DOGECOIN_HASH_LENGTH, 0);
80 if (!res)
81 return false;
82 dogecoin_hash(rnddata, DOGECOIN_HASH_LENGTH, hash);
83 unsigned char sig[74];
84 size_t siglen = 74;
85 if (!dogecoin_key_sign_hash(privkey, hash, sig, &siglen))
86 return false;
87 return dogecoin_pubkey_verify_sig(pubkey, hash, sig, siglen);
88}

90void dogecoin_privkey_encode_wif(const dogecoin_key* privkey, const dogecoin_chainparams* chain, char* privkey_wif, size_t* strsize_inout)
91{
92 uint8_t pkeybase58c[34];
93 pkeybase58c[0] = chain->b58prefix_secret_address;
94 pkeybase58c[33] = 1; /* always use compressed keys */
95 memcpy_safe(&pkeybase58c[1], privkey->privkey, DOGECOIN_ECKEY_PKEY_LENGTH);
96 assert(dogecoin_base58_encode_check(pkeybase58c, 34, privkey_wif, *strsize_inout) != 0);
97 dogecoin_mem_zero(&pkeybase58c, 34);
98}

100dogecoin_bool dogecoin_privkey_decode_wif(const char* privkey_wif, const dogecoin_chainparams* chain, dogecoin_key* privkey)
101{
102 if (!privkey_wif || strlen(privkey_wif) < 50)
103 return false;
104 const size_t privkey_len = strlen(privkey_wif);
105 uint8_t* privkey_data = (uint8_t*)dogecoin_calloc(1, privkey_len);
106 dogecoin_mem_zero(privkey_data, privkey_len);
107 size_t outlen = 0;
108 outlen = dogecoin_base58_decode_check(privkey_wif, privkey_data, privkey_len);
109 if (!outlen) {
110 dogecoin_free(privkey_data);
111 return false;
112 }
113 if (privkey_data[0] != chain->b58prefix_secret_address) {
114 dogecoin_free(privkey_data);
115 return false;
116 }
117 memcpy_safe(privkey->privkey, &privkey_data[1], DOGECOIN_ECKEY_PKEY_LENGTH);
118 dogecoin_mem_zero(privkey_data, sizeof(privkey_data));
119 dogecoin_free(privkey_data);
120 return true;
121}

152void dogecoin_pubkey_get_hash160(const dogecoin_pubkey* pubkey, uint160 hash160)
153{
154 uint256 hashout;
155 dogecoin_hash_sngl_sha256(pubkey->pubkey, pubkey->compressed ? DOGECOIN_ECKEY_COMPRESSED_LENGTH : DOGECOIN_ECKEY_UNCOMPRESSED_LENGTH, hashout);
156 rmd160(hashout, sizeof(hashout), hash160);
157}

197dogecoin_bool dogecoin_key_recover_pubkey(const unsigned char* sig, const uint256 hash, int recid, dogecoin_pubkey* pubkey)
198{
199 uint8_t pubkeybuf[128];
200 size_t outlen = 128;
201 if (!dogecoin_recover_pubkey(sig, hash, recid, pubkeybuf, &outlen) || outlen > DOGECOIN_ECKEY_UNCOMPRESSED_LENGTH)
202 return 0;
203 dogecoin_mem_zero(pubkey->pubkey, sizeof(pubkey->pubkey));
204 memcpy_safe(pubkey->pubkey, pubkeybuf, outlen);
205 if (outlen == DOGECOIN_ECKEY_COMPRESSED_LENGTH)
206 pubkey->compressed = true;
207 return 1;
208}

233dogecoin_bool init_keypair(char* privkeywif, dogecoin_key* key, dogecoin_pubkey* pubkey) {
234 dogecoin_privkey_init(key);
235 assert(dogecoin_privkey_is_valid(key) == 0);
236 const dogecoin_chainparams* chain = chain_from_b58_prefix(privkeywif);
237 if (!dogecoin_privkey_decode_wif(privkeywif, chain, key)) {
238 return false;
239 }
240 assert(dogecoin_privkey_is_valid(key) == 1);
241
242 dogecoin_pubkey_init(pubkey);
243 pubkey->compressed = false;
244 assert(dogecoin_pubkey_is_valid(pubkey) == 0);
245 dogecoin_pubkey_from_key(key, pubkey);
246 if (!dogecoin_pubkey_is_valid(pubkey)) return 0;
247 if (!dogecoin_privkey_verify_pubkey(key, pubkey)) return 0;
248 return true;
249}

252{
253 uint8_t hash160[sizeof(uint160) + 1];

255 dogecoin_pubkey_get_hash160(pubkey, hash160 + 1);
256 dogecoin_base58_encode_check(hash160, sizeof(hash160), addrout, 100);
257 return true;
