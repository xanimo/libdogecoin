52void dogecoin_tx_in_free(dogecoin_tx_in* tx_in)
53{

55 return;
56

59

62 tx_in->script_sig = NULL;

64
65 dogecoin_mem_zero(tx_in, sizeof(*tx_in));
66 dogecoin_free(tx_in);
67}

200int dogecoin_script_hash_to_p2pkh(dogecoin_tx_out* txout, char* p2pkh, int is_testnet) {
201 if (!txout) return false;
202
203 dogecoin_tx_out* copy = dogecoin_tx_out_new();
204 dogecoin_tx_out_copy(copy, txout);
205 size_t length = 2;
206 uint8_t* stripped_array = dogecoin_uint8_vla(txout->script_pubkey->len);
207 dogecoin_mem_zero(stripped_array, txout->script_pubkey->len * sizeof(stripped_array[0]));
208 // loop through 20 bytes of the script hash while stripping op codes
209 // and copy from index 2 to 21 after prefixing with version
210 // from chainparams:
211 for (; length < copy->script_pubkey->len - 4; length++) {
212 switch (copy->script_pubkey->str[length]) {
213 case OP_DUP:
214 break;
215 case (char)OP_HASH160:
216 break;
217 case (char)OP_EQUALVERIFY:
218 break;
219 case (char)OP_CHECKSIG:
220 break;
221 default:
222 copy->script_pubkey->str[2] = is_testnet ? 0x1e : 0x71;
223 memccpy(stripped_array, &copy->script_pubkey->str[2], 2, 21);
224 break;
225 }
226 }
227
228 unsigned char d1[SHA256_DIGEST_LENGTH], checksum[4], unencoded_address[25];
229 // double sha256 stripped array into d1:
230 dogecoin_dblhash((const unsigned char *)stripped_array, strlen((const char *)stripped_array), d1);
231 // copy check sum (4 bytes) into checksum var:
232 memcpy(checksum, d1, 4);
233 // copy stripped array into final var before passing to out variable:
234 memcpy(unencoded_address, stripped_array, 21);
235 free(stripped_array);
236 // copy checksum to the last 4 bytes of our unencoded_address:
237 unencoded_address[21] = checksum[0];
238 unencoded_address[22] = checksum[1];
239 unencoded_address[23] = checksum[2];
240 unencoded_address[24] = checksum[3];
241
242 char script_hash_to_p2pkh[35];
243 // base 58 encode check our unencoded_address into the script_hash_to_p2pkh:
244 if (!dogecoin_base58_encode_check(unencoded_address, 21, script_hash_to_p2pkh, sizeof(script_hash_to_p2pkh))) {
245 return false;
246 }
247
248 debug_print("doublesha: %s\n", utils_uint8_to_hex(d1, sizeof(d1)));
249 debug_print("checksum: %s\n", utils_uint8_to_hex(checksum, sizeof(checksum)));
250 debug_print("unencoded_address: %s\n", utils_uint8_to_hex(unencoded_address, sizeof(unencoded_address)));
251 debug_print("scripthash2p2pkh: %s\n", script_hash_to_p2pkh);
252
253 // copy to out variable p2pkh, free tx_out copy and return true:
254 memcpy(p2pkh, script_hash_to_p2pkh, sizeof(script_hash_to_p2pkh));
255 dogecoin_tx_out_free(copy);
256 return memcmp(p2pkh, script_hash_to_p2pkh, strlen(script_hash_to_p2pkh)) == 0;
257}

268char* dogecoin_p2pkh_to_script_hash(char* p2pkh) {
269 if (!p2pkh) return false;
270
271 // strlen(p2pkh) + 1 = 35
272 unsigned char dec[35]; //problem is here, it works if its char**
273
274 // MLUMIN: MSVC
275 size_t decoded_length = dogecoin_base58_decode_check(p2pkh, (uint8_t*)&dec, sizeof(dec) / sizeof(dec[0]));
276 if (decoded_length==0)
277 {
278 printf("failed base58 decode\n");
279 return false;
280 }
281
282 //decoded bytes = [1-byte versionbits][20-byte hash][4-byte checksum]
283 char* b58_decode_hex =utils_uint8_to_hex((const uint8_t*)dec, decoded_length - 4);
284
285 //2* (3-byte header + 20-byte hash + 2-byte footer) + 1-byte null terminator
286 char* tmp = dogecoin_malloc(40 + 6 + 4 + 1);
287
288 //concatenate the fields
289 sprintf(tmp, "%02x%02x%02x%.40s%02x%02x", OP_DUP, OP_HASH160, 20, &b58_decode_hex[2], OP_EQUALVERIFY, OP_CHECKSIG);
290 return tmp;
291}

305char* dogecoin_private_key_wif_to_script_hash(char* private_key_wif) {
306 if (!private_key_wif) {
307 return false;
308 }
309
310 const dogecoin_chainparams* chain = (private_key_wif[0] == 'c') ? &dogecoin_chainparams_test : &dogecoin_chainparams_main;
311
312 /* private key */
313 dogecoin_key key;
314 dogecoin_privkey_init(&key);
315 dogecoin_privkey_decode_wif(private_key_wif, chain, &key);
316 if (!dogecoin_privkey_is_valid(&key)) {
317 debug_print("private key is not valid!\nchain: %s\n", chain->chainname);
318 return false;
319 }
320
321 char new_wif_privkey[53];
322 size_t sizeout = sizeof(new_wif_privkey);
323 dogecoin_privkey_encode_wif(&key, chain, new_wif_privkey, &sizeout);
324
325 /* public key */
326 dogecoin_pubkey pubkey;
327 dogecoin_pubkey_init(&pubkey);
328 dogecoin_pubkey_from_key(&key, &pubkey);
329 if (!dogecoin_pubkey_is_valid(&pubkey)) {
330 debug_print("pubkey is not valid!\nchain: %s\n", chain->chainname);
331 return false;
332 }
333
334 char* new_p2pkh_pubkey = dogecoin_char_vla(sizeout);
335 dogecoin_pubkey_getaddr_p2pkh(&pubkey, chain, new_p2pkh_pubkey);
336 dogecoin_privkey_cleanse(&key);
337 dogecoin_pubkey_cleanse(&pubkey);
338 char* script_hash=dogecoin_p2pkh_to_script_hash(new_p2pkh_pubkey);
339 free(new_p2pkh_pubkey);
340 return script_hash;
341}

371dogecoin_bool dogecoin_tx_in_deserialize(dogecoin_tx_in* tx_in, struct const_buffer* buf)
372{
373 deser_u256(tx_in->prevout.hash, buf);
374 if (!deser_u32(&tx_in->prevout.n, buf)) {
375 return false;
376 }
377 if (!deser_varstr(&tx_in->script_sig, buf)) {
378 return false;
379 }
380 if (!deser_u32(&tx_in->sequence, buf)) {
381 return false;
382 }
383 return true;

419int dogecoin_tx_deserialize(const unsigned char* tx_serialized, size_t inlen, dogecoin_tx* tx, size_t* consumed_length)
420{
421 struct const_buffer buf = {tx_serialized, inlen};
422 if (consumed_length) {
423 *consumed_length = 0;
424 }
425
426 //tx needs to be initialized
427 deser_s32(&tx->version, &buf);
428
429 uint32_t vlen;
430 if (!deser_varlen(&vlen, &buf)) {
431 return false;
432 }
433
434 unsigned int i;
435 for (i = 0; i < vlen; i++) {
436 dogecoin_tx_in* tx_in = dogecoin_tx_in_new();
437
438 if (!dogecoin_tx_in_deserialize(tx_in, &buf)) {
439 dogecoin_tx_in_free(tx_in);
440 return false;
441 } else {
442 vector_add(tx->vin, tx_in);
443 }
444 }
445
446 if (!deser_varlen(&vlen, &buf))
447 return false;
448 for (i = 0; i < vlen; i++) {
449 dogecoin_tx_out* tx_out = dogecoin_tx_out_new();
450
451 if (!dogecoin_tx_out_deserialize(tx_out, &buf)) {
452 dogecoin_free(tx_out);
453 return false;
454 } else {
455 vector_add(tx->vout, tx_out);
456 }
457 }
458
459 if (!deser_u32(&tx->locktime, &buf)) {
460 return false;
461 }
462
463 if (consumed_length) {
464 *consumed_length = inlen - buf.len;
465 }
466 return true;
467}

510void dogecoin_tx_serialize(cstring* s, const dogecoin_tx* tx)
511{
512 ser_s32(s, tx->version);
513
514 ser_varlen(s, tx->vin ? tx->vin->len : 0);
515
516 unsigned int i;
517 if (tx->vin) {
518 for (i = 0; i < tx->vin->len; i++) {
519 dogecoin_tx_in* tx_in;
520
521 tx_in = vector_idx(tx->vin, i);
522 dogecoin_tx_in_serialize(s, tx_in);
523 }
524 }
525
526 ser_varlen(s, tx->vout ? tx->vout->len : 0);
527
528 if (tx->vout) {
529 for (i = 0; i < tx->vout->len; i++) {
530 dogecoin_tx_out* tx_out;
531
532 tx_out = vector_idx(tx->vout, i);
533 dogecoin_tx_out_serialize(s, tx_out);
534 }
535 }
536
537 ser_u32(s, tx->locktime);

618void dogecoin_tx_copy(dogecoin_tx* dest, const dogecoin_tx* src)
619{
620 dest->version = src->version;
621 dest->locktime = src->locktime;
622
623 if (!src->vin) {
624 dest->vin = NULL;
625 } else {
626 unsigned int i;
627
628 if (dest->vin != NULL) {
629 vector_free(dest->vin, true);
630 }
631
632 dest->vin = vector_new(src->vin->len, dogecoin_tx_in_free_cb);
633
634 for (i = 0; i < src->vin->len; i++) {
635 dogecoin_tx_in *tx_in_old, *tx_in_new;
636 tx_in_old = vector_idx(src->vin, i);
637 tx_in_new = dogecoin_malloc(sizeof(*tx_in_new));
638 dogecoin_tx_in_copy(tx_in_new, tx_in_old);
639 vector_add(dest->vin, tx_in_new);
640 }
641 }
642
643 if (!src->vout) {
644 dest->vout = NULL;
645 } else {
646 unsigned int i;
647
648 if (dest->vout)
649 vector_free(dest->vout, true);
650
651 dest->vout = vector_new(src->vout->len,
652 dogecoin_tx_out_free_cb);
653
654 for (i = 0; i < src->vout->len; i++) {
655 dogecoin_tx_out *tx_out_old, *tx_out_new;
656 tx_out_old = vector_idx(src->vout, i);
657 tx_out_new = dogecoin_malloc(sizeof(*tx_out_new));
658 dogecoin_tx_out_copy(tx_out_new, tx_out_old);
659 vector_add(dest->vout, tx_out_new);
660 }
661 }
662}

750dogecoin_bool dogecoin_tx_sighash(const dogecoin_tx* tx_to, const cstring* fromPubKey, size_t in_num, int hashtype, uint256 hash)
751{
752 if (in_num >= tx_to->vin->len || !tx_to->vout) {
753 return false;
754 }
755
756 dogecoin_bool ret = true;
757
758 dogecoin_tx* tx_tmp = dogecoin_tx_new();
759 dogecoin_tx_copy(tx_tmp, tx_to);
760
761 // standard sighash (SIGVERSION_BASE)
762 cstring* new_script = cstr_new_sz(fromPubKey->len);
763 dogecoin_script_copy_without_op_codeseperator(fromPubKey, new_script);
764
765 size_t i;
766 dogecoin_tx_in* tx_in;
767 for (i = 0; i < tx_tmp->vin->len; i++) {
768 tx_in = vector_idx(tx_tmp->vin, i);
769 cstr_resize(tx_in->script_sig, 0);
770 if (i == in_num) {
771 cstr_append_buf(tx_in->script_sig,
772 new_script->str,
773 new_script->len);
774 }
775 }
776 cstr_free(new_script, true);
777 if ((hashtype & 0x1f) == SIGHASH_NONE) {
778 if (tx_tmp->vout) {
779 vector_free(tx_tmp->vout, true);
780 }
781
782 tx_tmp->vout = vector_new(1, dogecoin_tx_out_free_cb);
783
784 for (i = 0; i < tx_tmp->vin->len; i++) {
785 tx_in = vector_idx(tx_tmp->vin, i);
786 if (i != in_num) {
787 tx_in->sequence = 0;
788 }
789 }
790 } else if ((hashtype & 0x1f) == SIGHASH_SINGLE) {
791 size_t n_out = in_num;
792 if (n_out >= tx_tmp->vout->len) {
793 //TODO: set error code
794 ret = false;
795 goto out;
796 }
797
798 vector_resize(tx_tmp->vout, n_out + 1);
799
800 for (i = 0; i < n_out; i++) {
801 dogecoin_tx_out* tx_out;
802
803 tx_out = vector_idx(tx_tmp->vout, i);
804 tx_out->value = -1;
805 if (tx_out->script_pubkey) {
806 cstr_free(tx_out->script_pubkey, true);
807 tx_out->script_pubkey = NULL;
808 }
809 }
810
811 for (i = 0; i < tx_tmp->vin->len; i++) {
812 tx_in = vector_idx(tx_tmp->vin, i);
813 if (i != in_num) {
814 tx_in->sequence = 0;
815 }
816 }
817 }
818
819 if (hashtype & SIGHASH_ANYONECANPAY) {
820 if (in_num > 0) {
821 vector_remove_range(tx_tmp->vin, 0, in_num);
822 }
823 vector_resize(tx_tmp->vin, 1);
824 }
825
826 cstring* s = cstr_new_sz(512);
827 dogecoin_tx_serialize(s, tx_tmp);
828 ser_s32(s, hashtype);
829 dogecoin_hash((const uint8_t*)s->str, s->len, hash);
830 cstr_free(s, true);
831
832out:
833 dogecoin_tx_free(tx_tmp);
834 return ret;
835}

907dogecoin_bool dogecoin_tx_add_address_out(dogecoin_tx* tx, const dogecoin_chainparams* chain, int64_t amount, const char* address)
908{
909 const size_t buflen = sizeof(uint8_t) * strlen(address) * 2;
910 uint8_t* buf = dogecoin_calloc(1, buflen);
911 size_t r = dogecoin_base58_decode_check(address, buf, buflen);
912 if (r > 0 && buf[0] == chain->b58prefix_pubkey_address) {
913 dogecoin_tx_add_p2pkh_hash160_out(tx, amount, &buf[1]);
914 } else if (r > 0 && buf[0] == chain->b58prefix_script_address) {

1064enum dogecoin_tx_sign_result dogecoin_tx_sign_input(dogecoin_tx* tx_in_out, const cstring* script, const dogecoin_key* privkey, size_t inputindex, int sighashtype, uint8_t* sigcompact_out, uint8_t* sigder_out, size_t* sigder_len_out)
1065{
1066 if (!tx_in_out || !script) {
1067 return DOGECOIN_SIGN_INVALID_TX_OR_SCRIPT;
1068 }
1069
1070 if (inputindex >= tx_in_out->vin->len) {
1071 return DOGECOIN_SIGN_INPUTINDEX_OUT_OF_RANGE;
1072 }
1073
1074 if (!dogecoin_privkey_is_valid(privkey)) {
1075 return DOGECOIN_SIGN_INVALID_KEY;
1076 }
1077
1078 // calculate pubkey
1079 dogecoin_pubkey pubkey;
1080 dogecoin_pubkey_init(&pubkey);
1081 dogecoin_pubkey_from_key(privkey, &pubkey);
1082 if (!dogecoin_pubkey_is_valid(&pubkey)) {
1083 return DOGECOIN_SIGN_INVALID_KEY;
1084 }
1085 enum dogecoin_tx_sign_result res = DOGECOIN_SIGN_OK;
1086
1087 cstring* script_sign = cstr_new_cstr(script); //copy the script because we may modify it
1088 dogecoin_tx_in* tx_in = vector_idx(tx_in_out->vin, inputindex);
1089 vector* script_pushes = vector_new(1, free);
1090
1091 enum dogecoin_tx_out_type type = dogecoin_script_classify(script, script_pushes);
1092 if (type == DOGECOIN_TX_PUBKEYHASH && script_pushes->len == 1) {
1093 // check if given private key matches the script
1094 uint160 hash160;
1095 dogecoin_pubkey_get_hash160(&pubkey, hash160);
1096 uint160* hash160_in_script = vector_idx(script_pushes, 0);
1097 if (memcmp(hash160_in_script, hash160, sizeof(hash160)) != 0) {
1098 res = DOGECOIN_SIGN_NO_KEY_MATCH; //sign anyways
1099 }
1100 } else {
1101 // unknown script, however, still try to create a signature (don't apply though)
1102 res = DOGECOIN_SIGN_UNKNOWN_SCRIPT_TYPE;
1103 }
1104 vector_free(script_pushes, true);
1105
1106 uint256 sighash;
1107 dogecoin_mem_zero(sighash, sizeof(sighash));
1108 if (!dogecoin_tx_sighash(tx_in_out, script_sign, inputindex, sighashtype, sighash)) {
1109 cstr_free(script_sign, true);
1110 return DOGECOIN_SIGN_SIGHASH_FAILED;
1111 }
1112 cstr_free(script_sign, true);
1113 // sign compact
1114 uint8_t sig[64];
1115 size_t siglen = 0;
1116 dogecoin_key_sign_hash_compact(privkey, sighash, sig, &siglen);
1117 assert(siglen == sizeof(sig));
1118 if (sigcompact_out) {
1119 memcpy_safe(sigcompact_out, sig, siglen);
1120 }
1121
1122 // form normalized DER signature & hashtype
1123 unsigned char sigder_plus_hashtype[74 + 1];
1124 size_t sigderlen = 75;
1125 dogecoin_ecc_compact_to_der_normalized(sig, sigder_plus_hashtype, &sigderlen);
1126 assert(sigderlen <= 74 && sigderlen >= 70);
1127 sigder_plus_hashtype[sigderlen] = sighashtype;
1128 sigderlen += 1; //+hashtype
1129 if (sigcompact_out) {
1130 memcpy_safe(sigder_out, sigder_plus_hashtype, sigderlen);
1131 }
1132 if (sigder_len_out) {
1133 *sigder_len_out = sigderlen;
1134 }
1135
1136 // apply signature depending on script type
1137 if (type == DOGECOIN_TX_PUBKEYHASH) {
1138 // apply DER sig
1139 ser_varlen(tx_in->script_sig, sigderlen);
1140 ser_bytes(tx_in->script_sig, sigder_plus_hashtype, sigderlen);
1141
1142 // apply pubkey
1143 ser_varlen(tx_in->script_sig, pubkey.compressed ? DOGECOIN_ECKEY_COMPRESSED_LENGTH : DOGECOIN_ECKEY_UNCOMPRESSED_LENGTH);
1144 ser_bytes(tx_in->script_sig, pubkey.pubkey, pubkey.compressed ? DOGECOIN_ECKEY_COMPRESSED_LENGTH : DOGECOIN_ECKEY_UNCOMPRESSED_LENGTH);
1145 } else {
1146 // append nothing
1147 res = DOGECOIN_SIGN_UNKNOWN_SCRIPT_TYPE;
1148 }
1149 return res;
1150}
