153dogecoin_bool dogecoin_random_bytes_internal(uint8_t* buf, uint32_t len, const uint8_t update_seed)
154 {
155#ifdef WIN32
156 (void)update_seed;
157 /* BCryptGenRandom, defined in <bcrypt.h>
158 <https://docs.microsoft.com/en-us/windows/win32/api/bcrypt/nf-bcrypt-bcryptgenrandom>
159 with the BCRYPT_USE_SYSTEM_PREFERRED_RNG flag
160 works in Windows 7 and newer. */
161 static int bcrypt_not_working /* = 0 */;
162 if (!bcrypt_not_working) {
163#if !HAVE_LIB_BCRYPT
164 if (!initialized)
165 initialize();
166#endif
167 if (BCryptGenRandomFunc != NULL
168 && BCryptGenRandomFunc(NULL, buf, len, BCRYPT_USE_SYSTEM_PREFERRED_RNG) == 0 /*STATUS_SUCCESS*/)
169 return 1;
170 bcrypt_not_working = 1;
171 }
172#if !HAVE_LIB_BCRYPT
173 /* CryptGenRandom, defined in <wincrypt.h>
174 <https://docs.microsoft.com/en-us/windows/win32/api/wincrypt/nf-wincrypt-cryptgenrandom>
175 works in older releases as well, but is now deprecated.
176 CryptAcquireContext, defined in <wincrypt.h>
177 <https://docs.microsoft.com/en-us/windows/win32/api/wincrypt/nf-wincrypt-cryptacquirecontexta> */
178 {
179 static int crypt_initialized /* = 0 */;
180 static HCRYPTPROV provider;
181 if (!crypt_initialized) {
182 if (CryptAcquireContextA(&provider, NULL, NULL, PROV_RSA_FULL, CRYPT_VERIFY_CONTEXT))
183 crypt_initialized = 1;
184 else
185 crypt_initialized = -1;
186 }
187 if (crypt_initialized >= 0) {
188 if (!CryptGenRandom(provider, len, buf)) {
189 errno = EIO;
190 return -1;
191 }
192 return 1;
193 }
194 }
195# endif
196 errno = ENOSYS;
197 return -1;
198#else
199 (void)update_seed; //unused
200 FILE* frand = fopen("/dev/urandom", "r"); // figure out why RANDOM_DEVICE is undeclared here
201 if (!frand)
202 return false;
203 size_t len_read = fread(buf, 1, len, frand);
204 assert(len_read == len);
205 fclose(frand);
206 return true;
207#endif
208 }
209#endif
