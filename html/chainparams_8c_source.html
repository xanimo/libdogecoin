115const dogecoin_chainparams* chain_from_b58_prefix(const char* address) {
116 /* determine address prefix for network chainparams */
117 uint8_t prefix[1];
118 memcpy(prefix, address, 1);
119 int count = 0;
120 switch (prefix[0]) {
121 case '9':
122 count++;
123 break;
124 case 'A':
125 count++;
126 break;
127 case 'd':
128 count++;
129 break;
130 case 'D':
131 count++;
132 break;
133 case 'Q':
134 count++;
135 break;
136 case '6':
137 count++;
138 break;
139 }
140 return count ? &dogecoin_chainparams_main : &dogecoin_chainparams_test;
141}
