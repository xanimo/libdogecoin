14dogecoin_bool dogecoin_ecc_start(void)
15{
16 dogecoin_random_init();
17 secp256k1_ctx = secp256k1_context_create(SECP256K1_CONTEXT_SIGN | SECP256K1_CONTEXT_VERIFY);
18 if (secp256k1_ctx == NULL)
19 return false;

21 int ret;
22 ret = dogecoin_random_bytes(seed, 32, 0);
