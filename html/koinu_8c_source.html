51const char* conversion_type_to_str(const enum conversion_type type)
52{
53 if (type == CONVERSION_SUCCESS) {
54 return "CONVERSION_SUCCESS";
55 } else if (type == CONVERSION_NON_DECIMAL) {
56 return "CONVERSION_NON_DECIMAL";
57 } else if (type == CONVERSION_INVALID_STR_TERMINATION) {
58 return "CONVERSION_INVALID_STR_TERMINATION";
59 } else if (type == CONVERSION_OUT_OF_RANGE) {
60 return "CONVERSION_OUT_OF_RANGE";

62 return "CONVERSION_OVERFLOW";

68 return "CONVERSION_FAILURE";

70 return "CONVERSION_UNKNOWN_ERROR";

74size_t check_length(char* string) {
75 // set max length for all string inputs to 20 to account for total supply in 2022
76 // (currently 132.67 billion dogecoin) + 1e8 koinu passing UINT64_MAX 184467440737
77 // in 9.854916426 years (2032) with output of 5,256,000,000 mined dogecoins per year
78 // TODO: in order to maintain portability on 64/32-bit platforms we can't support
79 // literal representations of numbers greater than UINT64_MAX: 18446744073709551615
80 // therefore within the next 9.854916426 years, we need to roll our own uint128_t
81 // capable of providing support greater than 20 digit unsigned integers needed to
82 // for uints in the trillions. once possible we need to implment max length below:
83 // set max length for all string inputs to 21 to account for total supply passing
84 // 1T in ~180 years from 2022. this limit will be valid for the next 1980 years so
85 // make sure to update in year 4002. :)
86 size_t integer_length = strlen(string);
87 if (integer_length > 22) return false;
88 else return integer_length;
89}

91enum conversion_type validate_conversion(uint64_t converted, const char* src, const char* src_end, const char* target_end) {
92 enum conversion_type type = CONVERSION_SUCCESS;
93 if (src && src_end && target_end) {
94 if (src_end == src) {
95 type = CONVERSION_NON_DECIMAL;
96 debug_print("%s: not a decimal\n", src);
97 }
98 if (*target_end != *src_end) {
99 type = CONVERSION_INVALID_STR_TERMINATION;
100 debug_print("%s: extra characters at end of input: %s\n", src, src_end);
101 }
102 }
103 if ((UINT64_MAX == converted) && ERANGE == errno) {
104 type = CONVERSION_OUT_OF_RANGE;
105 debug_print("%s out of range of type uint64_t\n", src);
106 }
107 if ((converted >= UINT64_MAX)) {
108 type = CONVERSION_OVERFLOW;
109 debug_print("%"PRIu64" greater than UINT64_MAX\n", converted);
110 }
111 if (converted == 0xFFFFF511543FB600) {
112 type = CONVERSION_UNDERFLOW;
113 debug_print("%"PRIu64" equal to 0xFFFFF511543FB600\n", converted);
114 }
115 return type;
116}

118int calc_length(uint64_t x) {
119 int count = 0;
120 while (x > 0) {
121 x /= 10;
122 count++;
123 }
124 return count;

127void string(uint64_t input, char output[]) {
128 uint64_t i = 0, n = input, remainder, length = 0;
129 while (n != 0) {
130 length++;
131 n /= 10;
132 }
133 for (; i < length; i++) {
134 remainder = input % 10;
135 input /= 10;
136 output[length - (i + 1)] = remainder + '0';
137 }
138 output[length] = '\0';
139}

141int koinu_to_coins_str(uint64_t koinu, char* str) {
142 enum conversion_type state = validate_conversion(koinu, NULL, NULL, NULL);
143 if (state != CONVERSION_SUCCESS) return false;
144
145 uint64_t i = 0, j =0, length = calc_length(koinu),
146 target = length < 9 ? 10 - length : length - 9;
147
148 if (length < 9) {
149 string(koinu, str);
150 size_t l = str ? strlen(str) : 0;
151 char* swap = dogecoin_char_vla(l + 1);
152 memcpy_safe(swap, str, l + 1);
153 for (; i < target; i++) {
154 if (i == 0 || i == 1) {
155 str[0] = '0';
156 str[1] = '.';
157 } else str[i] = '0';
158 }
159 for (; i < 10; i++, j++) str[i] = swap[j];
160 free(swap);
161 } else {
162 char tmp[21];
163 string(koinu, tmp);
164 for (; i < length; i++) {
165 if (i < target) str[i] = tmp[i];
166 else if (i == target) {
167 str[i] = tmp[i];
168 str[i + 1] = '.';
169 } else if (i > target) str[i + 1] = tmp[i];
170 }
171 str[length + 1] = '\0';
172 }
173
174 return true;
175}

177uint64_t coins_to_koinu_str(char* coins) {
178 if (coins[0] == '-') return false;
179 size_t length = check_length(coins);
180 if (!length) return false;
181
182 char *end, dogecoin_string[21], koinu_string[9];
183 dogecoin_mem_zero(dogecoin_string, 21);
184 dogecoin_mem_zero(koinu_string, 9);
185
186 size_t i = 0, j = 0, mantissa_length = 0;
187 for (; i < length; i++) {
188 dogecoin_string[i] = coins[i];
189 if (coins[i] == '.') {
190 j = i;
191 while (j < i + 8) {
192 dogecoin_string[j] = '0'; j++;
193 }
194 dogecoin_string[i] = '\0';
195 i++;
196 break;
197 }
198 }
199
200 for (j = 0; i < length && j < 8; i++, j++) {
201 koinu_string[j] = coins[i];
202 mantissa_length++;
203 }
204
205 if (mantissa_length != 8) {
206 for (i = mantissa_length; i < 8; i++) {
207 koinu_string[i] = '0';
208 }
209 } else koinu_string[strlen(koinu_string)] = '\0';
210
211 errno = 0;
212 uint64_t dogecoin = strtoull(dogecoin_string, &end, 10);
213 enum conversion_type state = validate_conversion(dogecoin, (const char*)dogecoin_string, end, "\0");
214 if (state == CONVERSION_SUCCESS) {
215 if (strlen(dogecoin_string) <= 12) dogecoin *= 100000000;
216 } else debug_print("%s\n", conversion_type_to_str(state));
217
218
219 errno = 0;
220 uint64_t koinu = strtoull(koinu_string, &end, 10) + dogecoin;
221 state = validate_conversion(koinu, (const char*)koinu_string, end, "\0");
222 if (state == CONVERSION_SUCCESS) return koinu;
223 else { debug_print("%s\n", conversion_type_to_str(state)); return false; }
224}
