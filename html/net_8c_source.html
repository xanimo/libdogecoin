120void read_cb(struct bufferevent* bev, void* ctx)
121{
122 struct evbuffer* input = bufferevent_get_input(bev);
123 if (!input)
124 return;
125
126 size_t length = evbuffer_get_length(input);
127 dogecoin_node* node = (dogecoin_node*)ctx;
128
129 if ((node->state & NODE_CONNECTED) != NODE_CONNECTED) {
130 // ignore messages from disconnected peers
131 return;
132 }
133 // expand the cstring buffer if required
134 cstr_alloc_minsize(node->recvBuffer, node->recvBuffer->len + length);
135
136 // copy direct to cstring, avoid another heap buffer
137 evbuffer_copyout(input, node->recvBuffer->str + node->recvBuffer->len, length);
138 node->recvBuffer->len += length;
139
140 // drain the event buffer
141 evbuffer_drain(input, length);
142
143 struct const_buffer buf = {node->recvBuffer->str, node->recvBuffer->len};
144 dogecoin_p2p_msg_hdr hdr;
145 char* read_upto = NULL;
146
147 do {
148 if (buf.len < DOGECOIN_P2P_HDRSZ) {
149 break;
150 }
151
152 dogecoin_p2p_deser_msghdr(&hdr, &buf);
153 if (hdr.data_len > DOGECOIN_MAX_P2P_MSG_SIZE) {
154 dogecoin_node_misbehave(node);
155 return;
156 }
157 if (buf.len < hdr.data_len) {
158 break;
159 }
160 if (buf.len >= hdr.data_len) {
161 struct const_buffer cmd_data_buf = {buf.p, buf.len};
162 if ((node->state & NODE_CONNECTED) != NODE_CONNECTED) {
163 return;
164 }
165 dogecoin_node_parse_message(node, &hdr, &cmd_data_buf);
166
167 buf.p = (const unsigned char*)buf.p + hdr.data_len;
168 buf.len -= hdr.data_len;
169
170 read_upto = (void*)buf.p;
171 }
172 if (buf.len == 0) {
173 node->recvBuffer->len = 0;
174 break;
175 }
176 } while (1);
177
178 if (read_upto != NULL && node->recvBuffer->len != 0 && read_upto != (node->recvBuffer->str + node->recvBuffer->len)) {
179 char* end = node->recvBuffer->str + node->recvBuffer->len;
180 size_t available_chunk_data = end - read_upto;
181 cstring* tmp = cstr_new_buf(read_upto, available_chunk_data);
182 cstr_free(node->recvBuffer, true);
183 node->recvBuffer = tmp;
184 }
185}

209#if defined(_WIN32) && defined(__x86_64__)
210void node_periodical_timer(long long int fd, short int event, void* ctx)
211#else
212void node_periodical_timer(int fd, short int event, void* ctx)
213#endif
214{
215 UNUSED(fd);
216 UNUSED(event);
217 dogecoin_node* node = (dogecoin_node*)ctx;
218 uint64_t now = time(NULL);
219
220 if (node->nodegroup->periodic_timer_cb)
221 if (!node->nodegroup->periodic_timer_cb(node, &now))
222 return;
223
224 if (node->time_started_con + DOGECOIN_CONNECT_TIMEOUT_S < now && ((node->state & NODE_CONNECTING) == NODE_CONNECTING)) {
225 node->state = 0;
226 node->time_started_con = 0;
227 node->state |= NODE_ERRORED;
228 node->state |= NODE_TIMEOUT;
229 dogecoin_node_connection_state_changed(node);
230 }
231
232 /* This is checking if the node is connected and if the last ping time is greater than the current
233 time plus the ping interval. */
234 if (((node->state & NODE_CONNECTED) == NODE_CONNECTED) && node->lastping + DOGECOIN_PING_INTERVAL_S < now) {
235 uint64_t nonce;
236 dogecoin_cheap_random_bytes((uint8_t*)&nonce, sizeof(nonce));
237 cstring* pingmsg = dogecoin_p2p_message_new(node->nodegroup->chainparams->netmagic, DOGECOIN_MSG_PING, &nonce, sizeof(nonce));
238 dogecoin_node_send(node, pingmsg);
239 cstr_free(pingmsg, true);
240 node->lastping = now;
241 }
242}

251void event_cb(struct bufferevent* ev, short type, void* ctx)
252{
253 UNUSED(ev);
254 dogecoin_node* node = (dogecoin_node*)ctx;
255 node->nodegroup->log_write_cb("Event callback on node %d\n", node->nodeid);
256
257 if (((type & BEV_EVENT_TIMEOUT) != 0) && ((node->state & NODE_CONNECTING) == NODE_CONNECTING)) {
258 node->nodegroup->log_write_cb("Timout connecting to node %d.\n", node->nodeid);
259 node->state = 0;
260 node->state |= NODE_ERRORED;
261 node->state |= NODE_TIMEOUT;
262 dogecoin_node_connection_state_changed(node);
263 } else if (((type & BEV_EVENT_EOF) != 0) ||
264 ((type & BEV_EVENT_ERROR) != 0)) {
265 node->state = 0;
266 node->state |= NODE_ERRORED;
267 node->state |= NODE_DISCONNECTED;
268 if ((type & BEV_EVENT_EOF) != 0) {
269 node->nodegroup->log_write_cb("Disconnected from the remote peer %d.\n", node->nodeid);
270 node->state |= NODE_DISCONNECTED_FROM_REMOTE_PEER;
271 }
272 else {
273 node->nodegroup->log_write_cb("Error connecting to node %d.\n", node->nodeid);
274 }
275 dogecoin_node_connection_state_changed(node);
276 } else if (type & BEV_EVENT_CONNECTED) {
277 node->nodegroup->log_write_cb("Successful connected to node %d.\n", node->nodeid);
278 node->state |= NODE_CONNECTED;
279 node->state &= ~NODE_CONNECTING;
280 node->state &= ~NODE_ERRORED;
281 dogecoin_node_connection_state_changed(node);
282 }
283 node->nodegroup->log_write_cb("Connected nodes: %d\n", dogecoin_node_group_amount_of_connected_nodes(node->nodegroup, NODE_CONNECTED));
284}

407dogecoin_node_group* dogecoin_node_group_new(const dogecoin_chainparams* chainparams)
408{
409 dogecoin_node_group* node_group;
410 node_group = dogecoin_calloc(1, sizeof(*node_group));
411#ifdef _WIN32
412 WSADATA wsaData;
413 WORD wVersionRequested;
414 int err;
415 /* Use the MAKEWORD(lowbyte, highbyte) macro declared in Windef.h */
416 wVersionRequested = MAKEWORD(2, 2);
417 err = WSAStartup(wVersionRequested, &wsaData);
418 printf("\ninitializing winsock...");
419 if (err != 0) {
420 /* Tell the user that we could not find a usable */
421 /* Winsock DLL. */
422 printf("WSAStartup failed. error code : %d", err);
423 }
424
425 printf("initialized.\n");
426 /* Confirm that the WinSock DLL supports 2.2.*/
427 /* Note that if the DLL supports versions greater */
428 /* than 2.2 in addition to 2.2, it will still return */
429 /* 2.2 in wVersion since that is the version we */
430 /* requested. */
431
432 if (LOBYTE(wsaData.wVersion) != 2 || HIBYTE(wsaData.wVersion) != 2) {
433 /* Tell the user that we could not find a usable */
434 /* WinSock DLL. */
435 printf("Could not find a usable version of Winsock.dll\n");
436 WSACleanup();
437 }
438 else
439 printf("winsock 2.2 dll was found okay\n");
440#endif
441 struct event_base *base = event_base_new();
442 node_group->event_base = base;
443 if (!base) {
444 dogecoin_free(node_group);
445 printf("node_group->event_base not created\n");
446 return NULL;
447 };
448
449 node_group->nodes = vector_new(1, dogecoin_node_free_cb);
450 node_group->chainparams = (chainparams ? chainparams : &dogecoin_chainparams_main);
451 node_group->parse_cmd_cb = NULL;
452 strcpy(node_group->clientstr, "libdogecoin 0.1");
453
454 /* nullify callbacks */
455 node_group->postcmd_cb = NULL;
456 node_group->node_connection_state_changed_cb = NULL;
457 node_group->should_connect_to_more_nodes_cb = NULL;
458 node_group->handshake_done_cb = NULL;
459 node_group->log_write_cb = net_write_log_null;
460 node_group->desired_amount_connected_nodes = 25;
461
462 return node_group;
463}

528int dogecoin_node_group_amount_of_connected_nodes(dogecoin_node_group* group, enum NODE_STATE state)
529{
530 int count = 0;
531 for (size_t i = 0; i < group->nodes->len; i++) {

533 if ((node->state & state) == state)

535 }

547dogecoin_bool dogecoin_node_group_connect_next_nodes(dogecoin_node_group* group)
548{
549 dogecoin_bool connected_at_least_to_one_node = false;
550 int connect_amount = group->desired_amount_connected_nodes - dogecoin_node_group_amount_of_connected_nodes(group, NODE_CONNECTED);
551 if (connect_amount <= 0)
552 return true;
553
554 connect_amount = connect_amount*3;
555 for (size_t i = 0; i < group->nodes->len; i++) {
556 dogecoin_node* node = vector_idx(group->nodes, i);
557 if (
558 !((node->state & NODE_CONNECTED) == NODE_CONNECTED) &&
559 !((node->state & NODE_CONNECTING) == NODE_CONNECTING) &&
560 !((node->state & NODE_DISCONNECTED) == NODE_DISCONNECTED) &&
561 !((node->state & NODE_ERRORED) == NODE_ERRORED)) {
562 /* setup buffer event */
563 node->event_bev = bufferevent_socket_new(group->event_base, -1, BEV_OPT_CLOSE_ON_FREE);
564 bufferevent_setcb(node->event_bev, read_cb, write_cb, event_cb, node);
565 bufferevent_enable(node->event_bev, EV_READ | EV_WRITE);
566 if (bufferevent_socket_connect(node->event_bev, (struct sockaddr*)&node->addr, sizeof(node->addr)) < 0) {
567 if (node->event_bev) {
568 bufferevent_free(node->event_bev);
569 node->event_bev = NULL;
570 }
571 return false;
572 }
573
574 /* setup periodic timer */
575 node->time_started_con = time(NULL);
576 struct timeval tv;
577 tv.tv_sec = DOGECOIN_PERIODICAL_NODE_TIMER_S;
578 tv.tv_usec = 0;
579 node->timer_event = event_new(group->event_base, 0, EV_TIMEOUT | EV_PERSIST, node_periodical_timer, node);
580 event_add(node->timer_event, &tv);
581 node->state |= NODE_CONNECTING;
582 connected_at_least_to_one_node = true;
583 node->nodegroup->log_write_cb("Trying to connect to %d...\n", node->nodeid);
584 connect_amount--;
585 if (connect_amount <= 0)
586 return true;
587 }
588 }
589 return connected_at_least_to_one_node;
590}

598void dogecoin_node_connection_state_changed(dogecoin_node* node)
599{
600 if (node->nodegroup->node_connection_state_changed_cb)
601 node->nodegroup->node_connection_state_changed_cb(node);
602
603 if ((node->state & NODE_ERRORED) == NODE_ERRORED) {
604 dogecoin_node_release_events(node);
605
606 /* connect to more nodes are required */
607 dogecoin_bool should_connect_to_more_nodes = true;
608 if (node->nodegroup->should_connect_to_more_nodes_cb)
609 should_connect_to_more_nodes = node->nodegroup->should_connect_to_more_nodes_cb(node);
610
611 if (should_connect_to_more_nodes && (dogecoin_node_group_amount_of_connected_nodes(node->nodegroup, NODE_CONNECTED) + dogecoin_node_group_amount_of_connected_nodes(node->nodegroup, NODE_CONNECTING) < node->nodegroup->desired_amount_connected_nodes))
612 dogecoin_node_group_connect_next_nodes(node->nodegroup);
613 }
614 if ((node->state & NODE_MISSBEHAVED) == NODE_MISSBEHAVED) {
615 if ((node->state & NODE_CONNECTED) == NODE_CONNECTED || (node->state & NODE_CONNECTING) == NODE_CONNECTING) {
616 dogecoin_node_disconnect(node);
617 }
618 } else
619 dogecoin_node_send_version(node);
620}

645void dogecoin_node_send_version(dogecoin_node* node)
646{
647 if (!node)
648 return;
649
650 cstring* version_msg_cstr = cstr_new_sz(256);
651
652 /* copy socket_addr to p2p addr */
653 dogecoin_p2p_address fromAddr;
654 dogecoin_p2p_address_init(&fromAddr);
655 dogecoin_p2p_address toAddr;
656 dogecoin_p2p_address_init(&toAddr);
657 dogecoin_addr_to_p2paddr(&node->addr, &toAddr);
658
659 /* create a version message struct */
660 dogecoin_p2p_version_msg version_msg;
661 dogecoin_mem_zero(&version_msg, sizeof(version_msg));
662
663 /* create a serialized version message */
664 dogecoin_p2p_msg_version_init(&version_msg, &fromAddr, &toAddr, node->nodegroup->clientstr, true);
665 dogecoin_p2p_msg_version_ser(&version_msg, version_msg_cstr);
666
667 /* create p2p message */
668 cstring* p2p_msg = dogecoin_p2p_message_new(node->nodegroup->chainparams->netmagic, DOGECOIN_MSG_VERSION, version_msg_cstr->str, version_msg_cstr->len);
669
670 /* send message */
671 dogecoin_node_send(node, p2p_msg);
672
673 /* cleanup */
674 cstr_free(version_msg_cstr, true);
675 cstr_free(p2p_msg, true);
676}

687int dogecoin_node_parse_message(dogecoin_node* node, dogecoin_p2p_msg_hdr* hdr, struct const_buffer* buf)
688{
689 node->nodegroup->log_write_cb("received command from node %d: %s\n", node->nodeid, hdr->command);
690 if (memcmp(hdr->netmagic, node->nodegroup->chainparams->netmagic, sizeof(node->nodegroup->chainparams->netmagic)) != 0) {
691 return dogecoin_node_misbehave(node);
692 }
693
694 /* send the header and buffer to the possible callback */
695 if (!node->nodegroup->parse_cmd_cb || node->nodegroup->parse_cmd_cb(node, hdr, buf)) {
696 if (strcmp(hdr->command, DOGECOIN_MSG_VERSION) == 0) {
697 dogecoin_p2p_version_msg v_msg_check;
698 if (!dogecoin_p2p_msg_version_deser(&v_msg_check, buf)) {
699 return dogecoin_node_misbehave(node);
700 }
701 if ((v_msg_check.services & DOGECOIN_NODE_NETWORK) != DOGECOIN_NODE_NETWORK) {
702 dogecoin_node_disconnect(node);
703 }
704 node->bestknownheight = v_msg_check.start_height;
705 node->nodegroup->log_write_cb("Connected to node %d: %s (%d)\n", node->nodeid, v_msg_check.useragent, v_msg_check.start_height);
706 /* confirm version via verack */
707 cstring* verack = dogecoin_p2p_message_new(node->nodegroup->chainparams->netmagic, DOGECOIN_MSG_VERACK, NULL, 0);
708 dogecoin_node_send(node, verack);
709 cstr_free(verack, true);
710 } else if (strcmp(hdr->command, DOGECOIN_MSG_VERACK) == 0) {
711 /* complete handshake if verack has been received */
712 node->version_handshake = true;
713 if (node->nodegroup->handshake_done_cb)
714 node->nodegroup->handshake_done_cb(node);
715 } else if (strcmp(hdr->command, DOGECOIN_MSG_PING) == 0) {
716 uint64_t nonce = 0;
717 if (!deser_u64(&nonce, buf)) {
718 return dogecoin_node_misbehave(node);
719 }
720 cstring* pongmsg = dogecoin_p2p_message_new(node->nodegroup->chainparams->netmagic, DOGECOIN_MSG_PONG, &nonce, 8);
721 dogecoin_node_send(node, pongmsg);
722 cstr_free(pongmsg, true);
723 }
724 }
725
726 if (node->nodegroup->postcmd_cb)
727 node->nodegroup->postcmd_cb(node, hdr, buf);
728
729 return true;
730}

743size_t dogecoin_get_peers_from_dns(const char* seed, vector* ips_out, int port, int family)
744{
745 if (!seed || !ips_out || (family != AF_INET && family != AF_INET6) || port > 99999) {
746 return 0;
747 }
748 char def_port[12] = {0};
749 sprintf(def_port, "%d", port);
750 struct evutil_addrinfo hints, *aiTrav = NULL, *aiRes = NULL;
751 dogecoin_mem_zero(&hints, sizeof(hints));
752 hints.ai_family = AF_INET;
753 hints.ai_socktype = SOCK_STREAM;
754 hints.ai_protocol = IPPROTO_TCP;
755
756 int err = evutil_getaddrinfo(seed, NULL, &hints, &aiRes);
757 if (err)
758 return 0;
759
760 aiTrav = aiRes;
761 while (aiTrav != NULL) {
762 int maxlen = 256;
763 char* ipaddr = dogecoin_calloc(1, maxlen);
764 if (aiTrav->ai_family == AF_INET) {
765 assert(aiTrav->ai_addrlen >= sizeof(struct sockaddr_in));
766 evutil_inet_ntop(aiTrav->ai_family, &((struct sockaddr_in*)(aiTrav->ai_addr))->sin_addr, ipaddr, maxlen);
767 }
768
769 if (aiTrav->ai_family == AF_INET6) {
770 assert(aiTrav->ai_addrlen >= sizeof(struct sockaddr_in6));
771 evutil_inet_ntop(aiTrav->ai_family, &((struct sockaddr_in6*)(aiTrav->ai_addr))->sin6_addr, ipaddr, maxlen);
772 }
773
774 memcpy_safe(ipaddr + strlen(ipaddr), ":", 1);

846static dogecoin_bool broadcast_timer_cb(dogecoin_node* node, uint64_t* now) {
847 struct broadcast_ctx* ctx = (struct broadcast_ctx*)node->nodegroup->ctx;
848
849 if (node->time_started_con > 0) {
850 node->nodegroup->log_write_cb("timer node %d, delta: %" PRIu64 " secs (timeout is: %d)\n", node->nodeid, (*now - ctx->start_time), ctx->timeout);
851 }
852 if ((*now - ctx->start_time) > ctx->timeout)
853 dogecoin_node_disconnect(node);

855 if ((node->hints & (1 << 1)) == (1 << 1)) {
856 dogecoin_node_disconnect(node);
857 }
858
859 if ((node->hints & (1 << 2)) == (1 << 2)) {
860 dogecoin_node_disconnect(node);
861 }
862
863 /* return true = run internal timer logic (ping, disconnect-timeout, etc.) */
864 return true;

873void broadcast_handshake_done(struct dogecoin_node_* node) {
874 char ipaddr[256];
875 struct sockaddr_in* ad = (struct sockaddr_in*)&node->addr;
876 evutil_inet_ntop(node->addr.sa_family, &ad->sin_addr, ipaddr, sizeof(ipaddr));
877
878 printf("Successfully connected to peer %d (%s)\n", node->nodeid, ipaddr);
879 struct broadcast_ctx* ctx = (struct broadcast_ctx*)node->nodegroup->ctx;
880 ctx->connected_to_peers++;
881
882 if (ctx->inved_to_peers >= ctx->max_peers_to_inv) {
883 return;
884 }
885
886 /* create a INV */
887 cstring* inv_msg_cstr = cstr_new_sz(256);
888 dogecoin_p2p_inv_msg inv_msg;
889 dogecoin_mem_zero(&inv_msg, sizeof(inv_msg));
890
891 uint256 hash;
892 dogecoin_tx_hash(ctx->tx, hash);
893 dogecoin_p2p_msg_inv_init(&inv_msg, DOGECOIN_INV_TYPE_TX, hash);
894
895 /* serialize the inv count (1) */
896 ser_varlen(inv_msg_cstr, 1);
897 dogecoin_p2p_msg_inv_ser(&inv_msg, inv_msg_cstr);
898
899 cstring* p2p_msg = dogecoin_p2p_message_new(node->nodegroup->chainparams->netmagic, DOGECOIN_MSG_INV, inv_msg_cstr->str, inv_msg_cstr->len);
900 cstr_free(inv_msg_cstr, true);
901 dogecoin_node_send(node, p2p_msg);
902 cstr_free(p2p_msg, true);
903
904 /* INV sent */
905 node->hints |= (1 << 0);
906 ctx->inved_to_peers++;
907 }

935void broadcast_post_cmd(struct dogecoin_node_* node, dogecoin_p2p_msg_hdr* hdr, struct const_buffer* buf) {
936 struct broadcast_ctx* ctx = (struct broadcast_ctx*)node->nodegroup->ctx;
937 if (strcmp(hdr->command, DOGECOIN_MSG_INV) == 0) {
938 /* hash the tx */
939 /* TODO: cache the hash */
940 uint256 hash;
941 dogecoin_tx_hash(ctx->tx, hash);
942
943 uint32_t vsize;
944 if (!deser_varlen(&vsize, buf)) {
945 dogecoin_node_misbehave(node);
946 return;
947 };
948 for (unsigned int i = 0; i < vsize; i++) {
949 dogecoin_p2p_inv_msg inv_msg;
950 if (!dogecoin_p2p_msg_inv_deser(&inv_msg, buf)) {
951 dogecoin_node_misbehave(node);
952 return;
953 }
954 if (memcmp(hash, inv_msg.hash, sizeof(hash)) == 0) {
955 /* tx found on peer */
956 node->hints |= (1 << 2);
957 printf("node %d has the tx\n", node->nodeid);
958 ctx->found_on_non_inved_peers++;
959 printf("tx successfully seen on node %d\n", node->nodeid);
960 }
961 }
962 }
963 else if (strcmp(hdr->command, DOGECOIN_MSG_GETDATA) == 0 && ((node->hints & (1 << 1)) != (1 << 1))) {
964 ctx->getdata_from_peers++;
965 // only allow a single object in GETDATA for the broadcaster
966 uint32_t vsize;
967 if (!deser_varlen(&vsize, buf) || vsize != 1) {
968 dogecoin_node_misbehave(node);
969 return;
970 }
971
972 dogecoin_p2p_inv_msg inv_msg;
973 dogecoin_mem_zero(&inv_msg, sizeof(inv_msg));
974 if (!dogecoin_p2p_msg_inv_deser(&inv_msg, buf) || inv_msg.type != DOGECOIN_INV_TYPE_TX) {
975 dogecoin_node_misbehave(node);
976 return;
977 };
978
979 /* send the tx */
980 cstring* tx_ser = cstr_new_sz(1024);
981 dogecoin_tx_serialize(tx_ser, ctx->tx);
982 cstring* p2p_msg = dogecoin_p2p_message_new(node->nodegroup->chainparams->netmagic, DOGECOIN_MSG_TX, tx_ser->str, tx_ser->len);
983 cstr_free(tx_ser, true);
984 dogecoin_node_send(node, p2p_msg);
985 cstr_free(p2p_msg, true);
986
987 /* tx sent */
988 node->hints |= (1 << 1);
989
990 printf("tx successfully sent to node %d\n", node->nodeid);
991 }
992 }

1006dogecoin_bool broadcast_tx(const dogecoin_chainparams* chain, const dogecoin_tx* tx, const char* ips, int maxpeers, int timeout, dogecoin_bool debug) {
1007 struct broadcast_ctx ctx;
1008 ctx.tx = tx;
1009 ctx.debuglevel = debug;
1010 ctx.timeout = timeout;
1011 ctx.max_peers_to_inv = 2;
1012 ctx.found_on_non_inved_peers = 0;
1013 ctx.getdata_from_peers = 0;
1014 ctx.inved_to_peers = 0;
1015 ctx.connected_to_peers = 0;
1016 ctx.max_peers_to_connect = maxpeers;
1017 /* create a node group */
1018 dogecoin_node_group* group = dogecoin_node_group_new(chain);
1019 group->desired_amount_connected_nodes = ctx.max_peers_to_connect;
1020 group->ctx = &ctx;
1021
1022 /* set the timeout callback */
1023 group->periodic_timer_cb = broadcast_timer_cb;
1024
1025 /* set a individual log print function */
1026 if (debug) {
1027 group->log_write_cb = net_write_log_printf;
1028 }
1029 group->postcmd_cb = broadcast_post_cmd;
1030 group->handshake_done_cb = broadcast_handshake_done;
1031 group->should_connect_to_more_nodes_cb = broadcast_should_connect_more;
1032
1033 dogecoin_node_group_add_peers_by_ip_or_seed(group, ips);
1034
1035 uint256 txhash;
1036 dogecoin_tx_hash(tx, txhash);
1037 char hexout[sizeof(txhash) * 2 + 1];
1038 utils_bin_to_hex(txhash, sizeof(txhash), hexout);
1039 hexout[sizeof(txhash) * 2] = 0;
1040 utils_reverse_hex(hexout, strlen(hexout));
1041 printf("Start broadcasting transaction: %s with timeout %d seconds\n", hexout, timeout);
1042 /* connect to the next node */
1043 ctx.start_time = time(NULL);
1044 printf("Trying to connect to nodes...\n");
1045 dogecoin_node_group_connect_next_nodes(group);
1046
1047 /* start the event loop */
1048 dogecoin_node_group_event_loop(group);
1049
1050 /* cleanup (free) nodes structures from the heap */
1051 dogecoin_node_group_free(group);
1052
1053 printf("\n\nResult:\n=============\n");
1054 printf("Max nodes to connect to: %d\n", ctx.max_peers_to_connect);
1055 printf("Successfully connected to nodes: %d\n", ctx.connected_to_peers);
1056 printf("Informed nodes: %d\n", ctx.inved_to_peers);
1057 printf("Requested from nodes: %d\n", ctx.getdata_from_peers);
1058 printf("Seen on other nodes: %d\n", ctx.found_on_non_inved_peers);
1059
1060 if (ctx.getdata_from_peers == 0)
1061 {
1062 printf("\nError: The transaction was not requested by the informed nodes. This usually happens when the transaction has already been broadcasted\n");
1063 }
1064 else if (ctx.found_on_non_inved_peers == 0)
1065 {
1066 printf("\nError: The transaction was not relayed back. Your transaction is very likely invalid (or was already broadcasted and picked up by an invalid node)\n");
1067 }
1068 return 1;
1069 }
