65int generatePrivPubKeypair(char* wif_privkey, char* p2pkh_pubkey, bool is_testnet)
66{
67 /* internal variables */
68
69 char wif_privkey_internal[WIF_UNCOMPRESSED_PRIVKEY_STRINGLEN]; //MLUMIN: Keylength (51 or 52 chars, depending on uncompressed or compressed) +1 for string termination 'internally'?
70 char p2pkh_pubkey_internal[P2PKH_ADDR_STRINGLEN]; //MLUMIN: no magic numbers. p2pkh address should be 34 characters, +1 though for string termination 'internally'?
71 size_t privkey_len = sizeof(wif_privkey_internal);
72
73
74 /* if nothing is passed in use internal variables */
75 if (wif_privkey) {
76 memcpy_safe(wif_privkey_internal, wif_privkey, sizeof(wif_privkey_internal));
77 }
78 if (p2pkh_pubkey) {
79 memcpy_safe(p2pkh_pubkey_internal, p2pkh_pubkey, sizeof(p2pkh_pubkey_internal));
80 }
81
82 /* determine if mainnet or testnet/regtest */
83 const dogecoin_chainparams* chain = is_testnet ? &dogecoin_chainparams_test : &dogecoin_chainparams_main;
84
85 /* generate a new private key */
86 dogecoin_key key;
87 dogecoin_privkey_init(&key);
88 dogecoin_privkey_gen(&key);
89 dogecoin_privkey_encode_wif(&key, chain, wif_privkey_internal, &privkey_len);
90
91 /* generate a new public key and export hex */
92 dogecoin_pubkey pubkey;
93 dogecoin_pubkey_init(&pubkey);
94 assert(dogecoin_pubkey_is_valid(&pubkey) == 0);
95 dogecoin_pubkey_from_key(&key, &pubkey);
96
97 /* export p2pkh_pubkey address */
98 dogecoin_pubkey_getaddr_p2pkh(&pubkey, chain, p2pkh_pubkey_internal);
99
100 if (wif_privkey) {
101 memcpy_safe(wif_privkey, wif_privkey_internal, sizeof(wif_privkey_internal));
102 }
103 if (p2pkh_pubkey) {
104 memcpy_safe(p2pkh_pubkey, p2pkh_pubkey_internal, sizeof(p2pkh_pubkey_internal));
105 }
106
107 /* reset internal variables */
108 dogecoin_mem_zero(wif_privkey_internal, strlen(wif_privkey_internal));
109 dogecoin_mem_zero(p2pkh_pubkey_internal, strlen(p2pkh_pubkey_internal));
110
111 /* cleanup memory */
112 dogecoin_pubkey_cleanse(&pubkey);
113 dogecoin_privkey_cleanse(&key);
114 return true;
115}

174int generateDerivedHDPubkey(const char* wif_privkey_master, char* p2pkh_pubkey)
175{
176 /* require master key */
177 if (!wif_privkey_master) {
178 return false;
179 }
180
181 /* determine address prefix for network chainparams */
182 const dogecoin_chainparams* chain = chain_from_b58_prefix(wif_privkey_master);
183
184 char str[P2PKH_ADDR_STRINGLEN];
185
186 /* if nothing is passed in use internal variables */
187 if (p2pkh_pubkey) {
188 memcpy_safe(str, p2pkh_pubkey, sizeof(str));
189 }
190
191 dogecoin_hdnode* node = dogecoin_hdnode_new();
192 dogecoin_hdnode_deserialize(wif_privkey_master, chain, node);
193
194 dogecoin_hdnode_get_p2pkh_address(node, chain, str, sizeof(str));
195
196 /* pass back to external variable if exists */
197 if (p2pkh_pubkey) {
198 memcpy_safe(p2pkh_pubkey, str, sizeof(str));
199 }
200
201 /* reset internal variables */
202 dogecoin_hdnode_free(node);
203 dogecoin_mem_zero(str, strlen(str));
204
205 return true;
206}

219int verifyPrivPubKeypair(char* wif_privkey, char* p2pkh_pubkey, bool is_testnet) {
220 /* require both private and public key */
221 if (!wif_privkey || !p2pkh_pubkey) return false;
222
223 /* set chain */
224 const dogecoin_chainparams* chain = is_testnet ? &dogecoin_chainparams_test : &dogecoin_chainparams_main;
225
226 /* verify private key */
227 dogecoin_key key;
228 dogecoin_privkey_init(&key);
229 dogecoin_privkey_decode_wif(wif_privkey, chain, &key);
230 if (!dogecoin_privkey_is_valid(&key)) return false;
231
232 char new_wif_privkey[WIF_UNCOMPRESSED_PRIVKEY_STRINGLEN];
233 size_t sizeout = sizeof(new_wif_privkey);
234 dogecoin_privkey_encode_wif(&key, chain, new_wif_privkey, &sizeout);
235
236 /* verify public key */
237 dogecoin_pubkey pubkey;
238 dogecoin_pubkey_init(&pubkey);
239 dogecoin_pubkey_from_key(&key, &pubkey);
240 if (!dogecoin_pubkey_is_valid(&pubkey)) return false;
241
242 /* verify address derived matches provided address */
243 char* new_p2pkh_pubkey = dogecoin_char_vla(sizeout);
244 dogecoin_pubkey_getaddr_p2pkh(&pubkey, chain, new_p2pkh_pubkey);
245 if (strcmp(p2pkh_pubkey, new_p2pkh_pubkey))
246 {
247 free(new_p2pkh_pubkey);
248 return false;
249 }
250
251 dogecoin_pubkey_cleanse(&pubkey);
252 dogecoin_privkey_cleanse(&key);
253 free(new_p2pkh_pubkey);
254 return true;
255}

296int verifyP2pkhAddress(char* p2pkh_pubkey, size_t len)
297{
298 if (!p2pkh_pubkey || !len) return false;
299 /* check length */
300 unsigned char* dec = dogecoin_uchar_vla(len);
301 unsigned char d1[SHA256_DIGEST_LENGTH];
302 unsigned char d2[SHA256_DIGEST_LENGTH];
303 if (!dogecoin_base58_decode_check(p2pkh_pubkey, dec, len))
304 {
305 free(dec);
306 return false;
307 }
308 /* check validity */
309 sha256_raw(dec, 21, d1);
310 sha256_raw(d1, SHA256_DIGEST_LENGTH, d2);
311 if (memcmp(dec + 21, d2, 4) != 0)
312 {
313 free(dec);
314 return false;
315 }
316 free(dec);
317 return true;
318}

333int getDerivedHDAddressByPath(const char* masterkey, const char* derived_path, char* outaddress, bool outprivkey) {
334 if (!masterkey || !derived_path || !outaddress) {
335 debug_print("%s", "missing input\n");
336 return false;
337 }
338
339 /* determine if mainnet or testnet/regtest */
340 const dogecoin_chainparams* chain = chain_from_b58_prefix(masterkey);
341 bool ret = true;
342 dogecoin_hdnode node, nodenew;
343
344 if (!dogecoin_hdnode_deserialize(masterkey, chain, &node)) {
345 ret = false;
346 }
347
348 // dogecoin_hdnode_has_privkey
349 bool pubckd = !dogecoin_hdnode_has_privkey(&node);
350 /* derive child key, use pubckd or privckd */
351 if (!dogecoin_hd_generate_key(&nodenew, derived_path, pubckd ? node.public_key : node.private_key, node.chain_code, pubckd)) {
352 ret = false;
353 }
354
355 if (outprivkey) dogecoin_hdnode_serialize_private(&nodenew, chain, outaddress, HD_MASTERKEY_STRINGLEN);
356 else dogecoin_hdnode_serialize_public(&nodenew, chain, outaddress, HD_MASTERKEY_STRINGLEN);
357 return ret;
358}

375int getDerivedHDAddress(const char* masterkey, uint32_t account, bool ischange, uint32_t addressindex, char* outaddress, bool outprivkey) {
376 if (!masterkey) {
377 debug_print("%s", "no extended key\n");
378 return false;
379 }
380
381 char derived_path[DERIVED_PATH_STRINGLEN];
382 int derived_path_size = snprintf(derived_path, sizeof(derived_path), "m/44'/3'/%u'/%u/%u", account, ischange, addressindex);
383
384 if (derived_path_size >= (int)sizeof(derived_path)) {
385 debug_print("%s", "derivation path overflow\n");
386 return false;
387 }
388
389 int ret = getDerivedHDAddressByPath(masterkey, derived_path, outaddress, outprivkey);
390 return ret;
391}

406int getDerivedHDAddressFromMnemonic(const uint32_t account, const uint32_t index, const CHANGE_LEVEL change_level, const MNEMONIC mnemonic, const PASSPHRASE pass, char* p2pkh_pubkey, const dogecoin_bool is_testnet) {
407
408 /* Validate input */
409 if (!mnemonic) {
410 fprintf(stderr, "ERROR: Invalid mnemonic\n");
411 return -1;
412 }
413
414 /* Initialize variables */
415 dogecoin_hdnode node;
416 dogecoin_hdnode bip44_key;
417 char keypath[BIP44_KEY_PATH_MAX_LENGTH + 1] = "";
418
419 /* Define seed and initialize */
420 SEED seed = {0};
421
422 /* Convert mnemonic to seed */
423 if (dogecoin_seed_from_mnemonic (mnemonic, pass, seed) == -1) {
424 return -1;
425 }
426
427 /* Generate the root key from the seed */
428 dogecoin_hdnode_from_seed(seed, MAX_SEED_SIZE, &node);
429
430 /* Derive the BIP 44 extended key */
431 if (derive_bip44_extended_private_key(&node, account, NULL, change_level, NULL, is_testnet, keypath, &bip44_key) == -1) {
432 return -1;
433 }
434
435 /* Derive the child private key at the index */
436 if (derive_bip44_extended_private_key(&node, account, &index, change_level, NULL, is_testnet, keypath, &bip44_key) == -1) {
437 return -1;
438 }
439
440 /* Generate the address */
441 dogecoin_hdnode_get_p2pkh_address(&bip44_key, is_testnet ? &dogecoin_chainparams_test : &dogecoin_chainparams_main, p2pkh_pubkey, P2PKH_ADDR_STRINGLEN);
442
443 return 0;
444
445}
