139void ser_varlen(cstring* s, uint32_t vlen)
140{
141 unsigned char c;
142
143 if (vlen < 253) {
144 c = vlen;
145 ser_bytes(s, &c, 1);
146 }
147
148 else if (vlen < 0x10000) {
149 c = 253;
150 ser_bytes(s, &c, 1);
151 ser_u16(s, (uint16_t)vlen);
152 }
153
154 else {
155 c = 254;
156 ser_bytes(s, &c, 1);
157 ser_u32(s, vlen);
158 }
159
160 /* u64 case intentionally not implemented */
161}

370int deser_varlen(uint32_t* lo, struct const_buffer* buf)
371{
372 uint32_t len;
373
374 unsigned char c;
375 if (!deser_bytes(&c, buf, 1))
376 return false;
377
378 if (c == 253) {
379 uint16_t v16;
380 if (!deser_u16(&v16, buf))
381 return false;
382 len = v16;
383 } else if (c == 254) {
384 uint32_t v32;
385 if (!deser_u32(&v32, buf))
386 return false;
387 len = v32;
388 } else if (c == 255) {
389 uint64_t v64;
390 if (!deser_u64(&v64, buf))
391 return false;
392 len = (uint32_t)v64; /* WARNING: truncate */
393 } else {
394 len = c;
395 }
396
397 *lo = len;
398 return true;
399}

413int deser_varlen_file(uint32_t* lo, FILE* file, uint8_t* rawdata, size_t* buflen_inout)
414{
415 uint32_t len;
416 struct const_buffer buf;
417 unsigned char c;
418 const unsigned char bufp[sizeof(uint64_t)];
419
420 /* check min size of the buffer */
421 if (*buflen_inout < sizeof(len))
422 return false;
423
424 if (fread(&c, 1, 1, file) != 1)
425 return false;
426
427 rawdata[0] = c;
428 *buflen_inout = 1;
429
430 buf.p = (void*)bufp;
431 buf.len = sizeof(uint64_t);
432
433 if (c == 253) {
434 uint16_t v16;
435 if (fread((void*)buf.p, 1, sizeof(v16), file) != sizeof(v16))
436 return false;
437 memcpy_safe(rawdata + 1, buf.p, sizeof(v16));
438 *buflen_inout += sizeof(v16);
439 if (!deser_u16(&v16, &buf))
440 return false;
441 len = v16;
442 } else if (c == 254) {
443 uint32_t v32;
444 if (fread((void*)buf.p, 1, sizeof(v32), file) != sizeof(v32))
445 return false;
446 memcpy_safe(rawdata + 1, buf.p, sizeof(v32));
447 *buflen_inout += sizeof(v32);
448 if (!deser_u32(&v32, &buf))
449 return false;
450 len = v32;
451 } else if (c == 255) {
452 uint64_t v64;
453 if (fread((void*)buf.p, 1, sizeof(v64), file) != sizeof(v64))
454 return false;
455 memcpy_safe(rawdata + 1, buf.p, sizeof(uint32_t)); /* warning, truncate! */
456 *buflen_inout += sizeof(uint32_t);
457 if (!deser_u64(&v64, &buf))
458 return false;
459 len = (uint32_t)v64; /* WARNING: truncate */
460 } else {
461 len = c;
462 }
463
464 *lo = len;
465 return true;
466}

479int deser_str(char* so, struct const_buffer* buf, size_t maxlen)
480{
481 uint32_t len;
482 uint32_t skip_len = 0;
483 if (!deser_varlen(&len, buf)) {
484 return false;
485 }
486
487 /* if input larger than buffer, truncate copy, skip remainder */
488 if (len > maxlen) {
489 skip_len = len - maxlen;
490 len = maxlen;
491 }
492
493 if (!deser_bytes(so, buf, len)) {
494 return false;
495 }
496 if (!deser_skip(buf, skip_len)) {
497 return false;
498 }
499
500 /* add C string null */
501 if (len < maxlen) {
502 so[len] = 0;
503 }
504 else {
505 so[maxlen - 1] = 0;
506 }
507
508 return true;
509}

521int deser_varstr(cstring** so, struct const_buffer* buf)
522{
523 uint32_t len;
524 cstring* s;
525 char* p;
526
527 if (*so) {
528 cstr_free(*so, 1);
529 *so = NULL;
530 }
531
532 if (!deser_varlen(&len, buf))
533 return false;
534
535 if (buf->len < len)
536 return false;
537
538 s = cstr_new_sz(len);
539 cstr_append_buf(s, buf->p, len);
540
541 p = (char*)buf->p;
542 p += len;
543 buf->p = p;
544 buf->len -= len;
545
546 *so = s;
547
548 return true;
549}
