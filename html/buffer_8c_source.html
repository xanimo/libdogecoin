61struct buffer* buffer_copy(const void* data, size_t data_len)
62{
63 struct buffer* buf;
64 buf = dogecoin_calloc(1, sizeof(*buf));
65 if (!buf) {
66 goto err_out;
67 }
68
69 buf->p = dogecoin_calloc(1, data_len);
70 if (!buf->p) {
71 goto err_out_free;
72 }
73
74 memcpy_safe(buf->p, data, data_len);
75 buf->len = data_len;
76
77 return buf;
78
79err_out_free:
80 dogecoin_free(buf);
81err_out:
82 return NULL;
83}
