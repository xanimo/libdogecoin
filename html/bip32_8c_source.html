142dogecoin_bool dogecoin_hdnode_from_seed(const uint8_t* seed, int seed_len, dogecoin_hdnode* out)
143{
144 uint8_t I[DOGECOIN_ECKEY_PKEY_LENGTH + DOGECOIN_BIP32_CHAINCODE_SIZE];
145 dogecoin_mem_zero(out, sizeof(dogecoin_hdnode));
146 out->depth = 0;
147 out->fingerprint = 0x00000000;
148 out->child_num = 0;
149 hmac_sha512((const uint8_t*)"Bitcoin seed", 12, seed, seed_len, I);
150 memcpy_safe(out->private_key, I, DOGECOIN_ECKEY_PKEY_LENGTH);
151
152 if (!dogecoin_ecc_verify_privatekey(out->private_key)) {
153 dogecoin_mem_zero(I, sizeof(I));
154 return false;
155 }
156
157 memcpy_safe(out->chain_code, I + DOGECOIN_ECKEY_PKEY_LENGTH, DOGECOIN_BIP32_CHAINCODE_SIZE);
158 dogecoin_hdnode_fill_public_key(out);
159 dogecoin_mem_zero(I, sizeof(I));
160 return true;
161}

173dogecoin_bool dogecoin_hdnode_public_ckd(dogecoin_hdnode* inout, uint32_t i)
174{
175 uint8_t data[1 + 32 + 4];
176 uint8_t I[32 + DOGECOIN_BIP32_CHAINCODE_SIZE];
177 uint8_t fingerprint[32];
178
179 if (i & 0x80000000) { // private derivation
180 return false;
181 } else { // public derivation
182 memcpy_safe(data, inout->public_key, DOGECOIN_ECKEY_COMPRESSED_LENGTH);
183 }
184 write_be(data + DOGECOIN_ECKEY_COMPRESSED_LENGTH, i);
185
186 sha256_raw(inout->public_key, DOGECOIN_ECKEY_COMPRESSED_LENGTH, fingerprint);
187 rmd160(fingerprint, 32, fingerprint);
188 inout->fingerprint = (fingerprint[0] << 24) + (fingerprint[1] << 16) + (fingerprint[2] << 8) + fingerprint[3];
189
190 dogecoin_mem_zero(inout->private_key, 32);
191
192 int failed = 0;
193 hmac_sha512(inout->chain_code, 32, data, sizeof(data), I);
194 memcpy_safe(inout->chain_code, I + 32, DOGECOIN_BIP32_CHAINCODE_SIZE);
195
196
197 if (!dogecoin_ecc_public_key_tweak_add(inout->public_key, I))
198 failed = false;
199
200 if (!failed) {
201 inout->depth++;
202 inout->child_num = i;
203 }
204
205 // Wipe all stack data.
206 dogecoin_mem_zero(data, sizeof(data));
207 dogecoin_mem_zero(I, sizeof(I));
208 dogecoin_mem_zero(fingerprint, sizeof(fingerprint));
209
210 return failed ? false : true;
211}

223dogecoin_bool dogecoin_hdnode_private_ckd(dogecoin_hdnode* inout, uint32_t i)
224{
225 uint8_t data[1 + DOGECOIN_ECKEY_PKEY_LENGTH + 4];
226 uint8_t I[DOGECOIN_ECKEY_PKEY_LENGTH + DOGECOIN_BIP32_CHAINCODE_SIZE];
227 uint8_t fingerprint[DOGECOIN_BIP32_CHAINCODE_SIZE];
228 uint8_t p[DOGECOIN_ECKEY_PKEY_LENGTH], z[DOGECOIN_ECKEY_PKEY_LENGTH];
229
230 if (i & 0x80000000) { // private derivation
231 data[0] = 0;
232 memcpy_safe(data + 1, inout->private_key, DOGECOIN_ECKEY_PKEY_LENGTH);
233 } else { // public derivation
234 memcpy_safe(data, inout->public_key, DOGECOIN_ECKEY_COMPRESSED_LENGTH);
235 }
236 write_be(data + DOGECOIN_ECKEY_COMPRESSED_LENGTH, i);
237
238 sha256_raw(inout->public_key, DOGECOIN_ECKEY_COMPRESSED_LENGTH, fingerprint);
239 rmd160(fingerprint, 32, fingerprint);
240 inout->fingerprint = (fingerprint[0] << 24) + (fingerprint[1] << 16) +
241 (fingerprint[2] << 8) + fingerprint[3];
242
243 dogecoin_mem_zero(fingerprint, sizeof(fingerprint));
244 memcpy_safe(p, inout->private_key, DOGECOIN_ECKEY_PKEY_LENGTH);
245
246 hmac_sha512(inout->chain_code, DOGECOIN_BIP32_CHAINCODE_SIZE, data, sizeof(data), I);
247 memcpy_safe(inout->chain_code, I + DOGECOIN_ECKEY_PKEY_LENGTH, DOGECOIN_BIP32_CHAINCODE_SIZE);
248 memcpy_safe(inout->private_key, I, DOGECOIN_ECKEY_PKEY_LENGTH);
249
250 memcpy_safe(z, inout->private_key, DOGECOIN_ECKEY_PKEY_LENGTH);
251
252 int failed = 0;
253 if (!dogecoin_ecc_verify_privatekey(z)) {
254 failed = 1;
255 }
256
257 memcpy_safe(inout->private_key, p, DOGECOIN_ECKEY_PKEY_LENGTH);
258 if (!dogecoin_ecc_private_key_tweak_add(inout->private_key, z)) {
259 failed = 1;
260 }
261
262 if (!failed) {
263 inout->depth++;
264 inout->child_num = i;
265 dogecoin_hdnode_fill_public_key(inout);
266 }
267
268 dogecoin_mem_zero(data, sizeof(data));
269 dogecoin_mem_zero(I, sizeof(I));
270 dogecoin_mem_zero(z, sizeof(z));
271 return true;
272}

303static void dogecoin_hdnode_serialize(const dogecoin_hdnode* node, uint32_t version, char use_public, char* str, size_t strsize)
304{

314 node_data[45] = 0;
315 memcpy_safe(node_data + 46, node->private_key, DOGECOIN_ECKEY_PKEY_LENGTH);
316 }
317 dogecoin_base58_encode_check(node_data, 78, str, strsize);
318}

426dogecoin_bool dogecoin_hdnode_deserialize(const char* str, const dogecoin_chainparams* chain, dogecoin_hdnode* node)
427{
428 if (!str || !chain || !node) return false;
429 const size_t ndlen = sizeof(uint8_t) * sizeof(dogecoin_hdnode);
430 uint8_t* node_data = (uint8_t*)dogecoin_calloc(1, ndlen);
431 dogecoin_mem_zero(node, sizeof(dogecoin_hdnode));
432 if (!dogecoin_base58_decode_check(str, node_data, ndlen)) {
433 dogecoin_free(node_data);
434 return false;
435 }
436 uint32_t version = read_be(node_data);
437 if (version == chain->b58prefix_bip32_pubkey) { // public node
438 memcpy_safe(node->public_key, node_data + 45, DOGECOIN_ECKEY_COMPRESSED_LENGTH);
439 } else if (version == chain->b58prefix_bip32_privkey) { // private node
440 if (node_data[45]) { // invalid data
441 dogecoin_free(node_data);
442 return false;
443 }
444 memcpy_safe(node->private_key, node_data + 46, DOGECOIN_ECKEY_PKEY_LENGTH);
445 dogecoin_hdnode_fill_public_key(node);
446 } else {
447 dogecoin_free(node_data);
448 return false; // invalid version
449 }
450 node->depth = node_data[4];
451 node->fingerprint = read_be(node_data + 5);
452 node->child_num = read_be(node_data + 9);
453 memcpy_safe(node->chain_code, node_data + 13, DOGECOIN_BIP32_CHAINCODE_SIZE);
454 dogecoin_free(node_data);
455 return true;
456}

471dogecoin_bool dogecoin_hd_generate_key(dogecoin_hdnode* node, const char* keypath, const uint8_t* keymaster, const uint8_t* chaincode, dogecoin_bool usepubckd)
472{
473 static char delim[] = "/";
474 static char prime[] = "phH\'";
475 static char digits[] = "0123456789";
476 uint64_t idx = 0;
477 assert(strlens(keypath) < 1024);
478 char *pch, *kp = dogecoin_malloc(strlens(keypath) + 1);
479 char *saveptr; /* for strtok_r calls - fix for concurrent calls error*/
480
481 if (!kp) {
482 return false;
483 }
484
485 if (strlens(keypath) < strlens("m/")) {
486 goto err;
487 }
488
489 dogecoin_mem_zero(kp, strlens(keypath) + 1);
490 memcpy_safe(kp, keypath, strlens(keypath));
491
492 if (kp[0] != 'm' || kp[1] != '/') {
493 goto err;
494 }
495
496 node->depth = 0;
497 node->child_num = 0;
498 node->fingerprint = 0;
499 memcpy_safe(node->chain_code, chaincode, DOGECOIN_BIP32_CHAINCODE_SIZE);
500 if (usepubckd == true) {
501 memcpy_safe(node->public_key, keymaster, DOGECOIN_ECKEY_COMPRESSED_LENGTH);
502 } else {
503 memcpy_safe(node->private_key, keymaster, DOGECOIN_ECKEY_PKEY_LENGTH);
504 dogecoin_hdnode_fill_public_key(node);
505 }
506
507 pch = strtok_r(kp + 2, delim, &saveptr);
508 while (pch != NULL) {
509 size_t i = 0;
510 int prm = 0;
511 for (; i < strlens(pch); i++) {
512 if (strchr(prime, pch[i])) {
513 if ((i != strlens(pch) - 1) || usepubckd == true) {
514 goto err;
515 }
516 prm = 1;
517 } else if (!strchr(digits, pch[i])) {
518 goto err;
519 }
520 }
521
522 idx = strtoull(pch, NULL, 10);
523 if (idx > UINT32_MAX) {
524 goto err;
525 }
526
527 if (prm) {
528 if (dogecoin_hdnode_private_ckd_prime(node, idx) != true) {
529 goto err;
530 }
531 } else {
532 if ((usepubckd == true ? dogecoin_hdnode_public_ckd(node, idx) : dogecoin_hdnode_private_ckd(node, idx)) != true) {
533 goto err;
534 }
535 }
536 pch = strtok_r(NULL, delim, &saveptr);
537 }
538 dogecoin_free(kp);
539 return true;
540
541err:
542 dogecoin_free(kp);
543 return false;
544}
