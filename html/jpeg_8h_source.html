50/*
51 * Create a JPEG encoder with default quality factor
52 * `img' specifies the pointer to aligned image data.
53 * `w' specifies the image width in pixels.
54 * `h' specifies the image height in pixels.
55 * Because the returned encoder is allocated by this function, it should be
56 * released with the `jpec_enc_del' call when it is no longer useful.
57 * Note: for efficiency the image data is *NOT* copied and the encoder just
58 * retains a pointer to it. Thus the image data must not be deleted
59 * nor change until the encoder object gets deleted.
60 */
61jpec_enc_t* jpec_enc_new(const uint8_t* img, uint16_t w, uint16_t h);
62/*
63 * `q` specifies the JPEG quality factor in 0..100
64 */
65jpec_enc_t* jpec_enc_new2(const uint8_t* img, uint16_t w, uint16_t h, int q);
66
67/*
68 * Release a JPEG encoder object
69 * `e` specifies the encoder object
70 */
71void jpec_enc_del(jpec_enc_t* e);
72
73/*
74 * Run the JPEG encoding
75 * `e` specifies the encoder object
76 * `len` specifies the pointer to the variable into which the length of the
77 * JPEG blob is assigned
78 * If successful, the return value is the pointer to the JPEG blob. `NULL` is
79 * returned if an error occurred.
80 * Note: the caller should take care to copy or save the JPEG blob before
81 * calling `jpec_enc_del` since the blob will no longer be maintained after.
82 */
83const uint8_t* jpec_enc_run(jpec_enc_t* e, int* len);
