22static int cstr_alloc_min_sz(cstring* s, size_t sz)
23{
24 unsigned int shift;
25 unsigned int al_sz;
26 char* new_s;
27
28 sz++; /* NULL overhead */
29
30 if (s->alloc && (s->alloc >= sz)) {
31 return 1;
32 }
33
34 shift = 3;
35 while ((al_sz = (1 << shift)) < sz) {
36 shift++;
37 }
38
39 new_s = dogecoin_realloc(s->str, al_sz);
40 if (!new_s) {
41 return 0;
42 }
43
44 s->str = new_s;
45 s->alloc = al_sz;
46 s->str[s->len] = 0;
47
48 return 1;
49}

97cstring* cstr_new_sz(size_t sz)
98{
99 cstring* s = dogecoin_calloc(1, sizeof(cstring));
100 if (!s) {

288int cstr_equal(const cstring* a, const cstring* b)

313int cstr_compare(const cstring* a, const cstring* b)
314{
315 unsigned int i;
316 if (a->len > b->len) {
317 return (1);
318 }
319 if (a->len < b->len) {
320 return (-1);
321 }
322
323 /* length equal, byte per byte compare */
324 for (i = 0; i < a->len; i++) {
325 char a1 = a->str[i];
326 char b1 = b->str[i];
327
328 if (a1 > b1) {
329 return (1);
330 }
331 if (a1 < b1) {
332 return (-1);
333 }
334 }
335 return (0);
336}

349int cstr_erase(cstring* s, size_t pos, ssize_t len)
350{
351 ssize_t old_tail;
352
353 if (pos == s->len && len == 0) {
354 return 1;
355 }
356 if (pos >= s->len) {
357 return 0;
358 }
359
360 old_tail = s->len - pos;
361 if ((len >= 0) && (len > old_tail)) {
362 return 0;
363 }
364
365 memmove(&s->str[pos], &s->str[pos + len], old_tail - len);
366 s->len -= len;
367 s->str[s->len] = 0;
368
369 return 1;
370}
