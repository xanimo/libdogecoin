52int dogecoin_base58_decode(void* bin, size_t* binszp, const char* b58, size_t b58sz)
53{
54 size_t binsz = *binszp;
55 if (binsz == 0)
56 return false;
57 const unsigned char* b58u = (const unsigned char*)b58;
58 unsigned char* binu = bin;
59 size_t outisz = (binsz + sizeof(b58_almostmaxint_t) - 1) / sizeof(b58_almostmaxint_t);
60 b58_maxint_t t;
61 b58_almostmaxint_t c;
62 size_t i, j = 0;
63 uint8_t bytesleft = binsz % sizeof(b58_almostmaxint_t);
64 b58_almostmaxint_t* outi = dogecoin_uint32_vla(outisz);
65 uint32_t zeromask = bytesleft ? (b58_almostmaxint_mask << (bytesleft * 8)) : 0;
66 unsigned zerocount = 0;
67
68 dogecoin_mem_zero(outi, outisz * sizeof(*outi));
69 for (i = 0; i < b58sz && b58u[i] == '1'; ++i) {
70 ++zerocount; // leading zeros, just count
71 }
72
73 for (; i < b58sz; ++i) {
74 if (b58u[i] & 0x80) {
75 free(outi);
76 return false; // high-bit set on invalid digit
77 }
78 if (b58digits_map[b58u[i]] == -1) {
79 free(outi);
80 return false; // invalid base58 digit
81 }
82 c = (unsigned)b58digits_map[b58u[i]];
83 for (j = outisz; --j;) {
84 t = ((b58_maxint_t)outi[j]) * 58 + c;
85 c = t >> b58_almostmaxint_bits;
86 outi[j] = t & b58_almostmaxint_mask;
87 }
88 if (c) {
89 dogecoin_mem_zero(outi, outisz * sizeof(*outi));
90 free(outi);
91 return false; // output number too big (carry to the next int32)
92 }
93 if (outi[0] & zeromask) {
94 dogecoin_mem_zero(outi, outisz * sizeof(*outi));
95 free(outi);
96 return false; // output number too big (last int32 filled too far)
97 }
98 }
99 j = 0;
100 if (bytesleft) {
101 for (i = bytesleft; i > 0; --i) {
102 *(binu++) = (outi[0] >> (8 * (i - 1))) & 0xff;
103 }
104 ++j;
105 }
106 for (; j < outisz; ++j) {
107 for (i = sizeof(*outi); i > 0; --i) {
108 *(binu++) = (outi[j] >> (8 * (i - 1))) & 0xff;
109 }
110 }
111 // count canonical base58 byte count
112 binu = bin; // locate the most significant byte
113 for (i = 0; i < binsz; ++i) {
114 if (binu[i]) break;
115 --*binszp;
116 }
117 // prepend the correct number of null-bytes
118 if (zerocount > i) {
119 free(outi);
120 return false; /* result too large */
121 }
122 *binszp += zerocount;
123 dogecoin_mem_zero(outi, outisz * sizeof(*outi));
124 free(outi);
125 return true;
126}

155int dogecoin_base58_encode(char* b58, size_t* b58sz, const void* data, size_t binsz)
156{
157 const uint8_t* bin = data;
158 int carry = 0;
159 ssize_t i = 0, j = 0, high = 0, zcount = 0;
160 size_t size = 0;
161 while (zcount < (ssize_t)binsz && !bin[zcount]) {
162 ++zcount;
163 }
164 size = (binsz - zcount) * 138 / 100 + 1;
165 uint8_t* buf = dogecoin_uint8_vla(size);
166 dogecoin_mem_zero(buf, size);
167 for (i = zcount, high = size - 1; i < (ssize_t)binsz; ++i, high = j) {
168 for (carry = bin[i], j = size - 1; (j > high) || carry; --j) {
169 carry += 256 * buf[j];
170 buf[j] = carry % 58;
171 carry /= 58;
172 if (!j) // Otherwise j wraps to maxint which is > high
173 break;
174 }
175 }
176 for (j = 0; j < (ssize_t)size && !buf[j]; ++j)
177 ;
178 if (*b58sz <= zcount + size - j) {
179 *b58sz = zcount + size - j + 1;
180 dogecoin_mem_zero(buf, size);
181 free(buf);
182 return false;
183 }
184 if (zcount) {
185 memset(b58, '1', zcount);
186 }
187 for (i = zcount; j < (ssize_t)size; ++i, ++j) {
188 b58[i] = b58digits_ordered[buf[j]];
189 }
190 b58[i] = '\0';
191 *b58sz = i + 1;
192 dogecoin_mem_zero(buf, size);
193 free(buf);
194 return true;
195}
