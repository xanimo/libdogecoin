42vector* vector_new(size_t res, void (*free_f)(void*))
43{
44 vector* vec = dogecoin_calloc(1, sizeof(vector));
45 if (!vec)
46 return NULL;
47
48 vec->alloc = 8;
49 while (vec->alloc < res)
50 vec->alloc *= 2;
51
52 vec->elem_free_f = free_f;
53 vec->data = dogecoin_calloc(1, vec->alloc * sizeof(void*));
54 if (!vec->data) {
55 dogecoin_free(vec);
56 return NULL;
57 }
58
59 return vec;
60}

73static void vector_free_data(vector* vec)
74{
75 if (!vec->data)
76 return;
77
78 if (vec->elem_free_f) {
79 unsigned int i;
80 for (i = 0; i < vec->len; i++)
81 if (vec->data[i]) {
82 vec->elem_free_f(vec->data[i]);
83 vec->data[i] = NULL;
84 }
85 }
86
87 dogecoin_free(vec->data);
88 vec->data = NULL;
89 vec->alloc = 0;
90 vec->len = 0;
91}

127static dogecoin_bool vector_grow(vector* vec, size_t min_sz)
128{
129 size_t new_alloc = vec->alloc;
130 while (new_alloc < min_sz) {
131 new_alloc *= 2;
132 }
133
134 if (vec->alloc == new_alloc) {
135 return true;
136 }
137
138 void* new_data = dogecoin_realloc(vec->data, new_alloc * sizeof(void*));
139 if (!new_data) {
140 return false;
141 }
142
143 vec->data = new_data;
144 vec->alloc = new_alloc;
145 return true;
146}

182dogecoin_bool vector_add(vector* vec, void* data)
183{
184 if (vec->len == vec->alloc) {
185 if (!vector_grow(vec, vec->len + 1)) {
186 return false;
187 }
188 }
189
190 vec->data[vec->len] = data;
191 vec->len++;
192 return true;

206void vector_remove_range(vector* vec, size_t pos, size_t len)
207{
208 if (!vec || ((pos + len) > vec->len)) {
209 return;
210 }
211
212 if (vec->elem_free_f) {
213 size_t i, count;
214 for (i = pos, count = 0; count < len; i++, count++) {
215 vec->elem_free_f(vec->data[i]);
216 }
217 }
218
219 memmove(&vec->data[pos], &vec->data[pos + len], (vec->len - pos - len) * sizeof(void*));
220 vec->len -= len;
221}

271dogecoin_bool vector_resize(vector* vec, size_t newsz)
272{
273 size_t i;
274
275 /* same size */
276 if (newsz == vec->len) {
277 return true;
278 }
279
280 /* truncate */
281 else if (newsz < vec->len) {
282 size_t del_count = vec->len - newsz;
283
284 for (i = (vec->len - del_count); i < vec->len; i++) {
285 if (vec->elem_free_f) {
286 vec->elem_free_f(vec->data[i]);
287 }
288 vec->data[i] = NULL;
289 }
290
291 vec->len = newsz;
292 return true;
293 }
294
295 /* last possibility: grow */
296 if (!vector_grow(vec, newsz)) {
297 return false;
298 }
299
300 /* set new elements to NULL */
301 for (i = vec->len; i < newsz; i++) {
302 vec->data[i] = NULL;
303 }
304
305 return true;
306}
