76/* support substitute for GNU only tdestroy */
77/* let's hope the node struct is always compatible */
78
79struct dogecoin_btree_node {
80 void* key;
81 struct dogecoin_btree_node* left;
82 struct dogecoin_btree_node* right;
83};
84
85static inline void dogecoin_btree_tdestroy(void* root, void (*freekey)(void*))
86{
87 struct dogecoin_btree_node* r = (struct dogecoin_btree_node*)root;
88 if (r == 0) {
89 return;
90 }
91 dogecoin_btree_tdestroy(r->left, freekey);
92 dogecoin_btree_tdestroy(r->right, freekey);
93 if (freekey) {
94 freekey(r->key);
95 }
96 dogecoin_free(r);
97}
