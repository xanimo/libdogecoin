59enum service_bits {
60 // Nothing
61 DOGECOIN_NODE_NONE = 0,
62 // NODE_NETWORK means that the node is capable of serving the block chain. It is currently
63 // set by all Bitcoin Core nodes, and is unset by SPV clients or other peers that just want
64 // network services but don't provide them.
65 DOGECOIN_NODE_NETWORK = (1 << 0),
66 // NODE_GETUTXO means the node is capable of responding to the getutxo protocol request.
67 // Bitcoin Core does not support this but a patch set called Bitcoin XT does.
68 // See BIP 64 for details on how this is implemented.
69 DOGECOIN_NODE_GETUTXO = (1 << 1),
70 // NODE_BLOOM means the node is capable and willing to handle bloom-filtered connections.
71 // Bitcoin Core nodes used to support this by default, without advertising this bit,
72 // but no longer do as of protocol version 70011 (= NO_BLOOM_VERSION)
73 DOGECOIN_NODE_BLOOM = (1 << 2),
74 // NODE_XTHIN means the node supports Xtreme Thinblocks
75 // If this is turned off then the node will not service nor make xthin requests
76 DOGECOIN_NODE_XTHIN = (1 << 4),
77 DOGECOIN_NODE_COMPACT_FILTERS = (1 << 6),
78};
