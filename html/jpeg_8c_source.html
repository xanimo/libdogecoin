86void jpec_buffer_write_byte(jpec_buffer_t* b, int val) {
87 assert(b);
88 if (b->siz == b->len) {
89 int nsiz = (b->siz > 0) ? 2 * b->siz : JPEC_BUFFER_INIT_SIZ;
90 void* tmp = realloc(b->stream, nsiz);
91 b->stream = (uint8_t*)tmp;
92 b->siz = nsiz;
93 }
94 b->stream[b->len++] = (uint8_t)val;
95}

148static void jpec_huff_encode_block_impl(jpec_block_t* block, jpec_huff_state_t* s) {
149 assert(block && s);
150 int val, bits, nbits;
151 /* DC coefficient encoding */
152 if (block->len > 0) {
153 val = block->zz[0] - s->dc;
154 s->dc = block->zz[0];
155 }
156 else {
157 val = -s->dc;
158 s->dc = 0;
159 }
160 bits = val;
161 if (val < 0) {
162 val = -val;
163 bits = ~val;
164 }
165 JPEC_HUFF_NBITS(nbits, val);
166 jpec_huff_write_bits(s, jpec_dc_code[nbits], jpec_dc_len[nbits]);
167 if (nbits) jpec_huff_write_bits(s, (unsigned int)bits, nbits);
168 /* AC coefficients encoding (w/ RLE of zeros) */
169 int nz = 0;
170 for (int i = 1; i < block->len; i++) {
171 if ((val = block->zz[i]) == 0) nz++;
172 else {
173 while (nz >= 16) {
174 jpec_huff_write_bits(s, jpec_ac_code[0xF0], jpec_ac_len[0xF0]); /* ZRL code */
175 nz -= 16;
176 }
177 bits = val;
178 if (val < 0) {
179 val = -val;
180 bits = ~val;
181 }
182 JPEC_HUFF_NBITS(nbits, val);
183 int j = (nz << 4) + nbits;
184 jpec_huff_write_bits(s, jpec_ac_code[j], jpec_ac_len[j]);
185 if (nbits) jpec_huff_write_bits(s, (unsigned int)bits, nbits);
186 nz = 0;
187 }
188 }
189 if (block->len < 64) {
190 jpec_huff_write_bits(s, jpec_ac_code[0x00], jpec_ac_len[0x00]); /* EOB marker */
191 }
192}

194/* Write n bits into the JPEG buffer, with 0 < n <= 16.
195 *
196 * == Details
197 * - 16 bits are large enough to hold any zig-zag coeff or the longest AC code
198 * - bits are chunked into bytes before being written into the JPEG buffer
199 * - any remaining bits are kept in memory by the Huffman state
200 * - at most 7 bits can be kept in memory
201 * - a 32-bit integer buffer is used internally
202 * - only the right 24 bits part of this buffer are used
203 * - the input bits and remaining bits (if any) are left-justified on this part
204 * - a mask is used to mask off any extra bits: useful when the input value has been
205 * first transformed by bitwise complement(|initial value|)
206 * - if an 0xFF byte is detected a 0x00 stuff byte is automatically written right after
207 */
208static void jpec_huff_write_bits(jpec_huff_state_t* s, unsigned int bits, int n) {
209 assert(s && n > 0 && n <= 16);
210 int32_t mask = (((int32_t)1) << n) - 1;
211 int32_t buffer = (int32_t)bits;
212 int nbits = s->nbits + n;
213 buffer &= mask;
214 buffer <<= 24 - nbits;
215 buffer |= s->buffer;
216 while (nbits >= 8) {
217 int chunk = (int)((buffer >> 16) & 0xFF);
218 jpec_buffer_write_byte(s->buf, chunk);
219 if (chunk == 0xFF) jpec_buffer_write_byte(s->buf, 0x00);
220 buffer <<= 8;
221 nbits -= 8;
222 }
223 s->buffer = buffer;
224 s->nbits = nbits;
225}

378jpec_enc_t* jpec_enc_new2(const uint8_t* img, uint16_t w, uint16_t h, int q) {
379 if (img && w > 0 && !(w & 0x7) && h > 0 && !(h & 0x7))
380 {
381 printf("width and height assert failed\n");
382 }
383 jpec_enc_t* e = malloc(sizeof(*e));
384 e->img = img;
385 e->w = w;
386 e->w8 = (((w - 1) >> 3) + 1) << 3;
387 e->h = h;
388 e->qual = q;
389 e->bmax = (((w - 1) >> 3) + 1) * (((h - 1) >> 3) + 1);
390 e->bnum = -1;
391 e->bx = -1;
392 e->by = -1;
393 int bsiz = JPEC_ENC_HEAD_SIZ + e->bmax * JPEC_ENC_BLOCK_SIZ;
394 e->buf = jpec_buffer_new2(bsiz);
395 e->hskel = malloc(sizeof(*e->hskel));
396 return e;
397}

406const uint8_t* jpec_enc_run(jpec_enc_t* e, int* len) {
407 assert(e && len);
408 jpec_enc_open(e);
409 while (jpec_enc_next_block(e)) {
410 jpec_enc_block_dct(e);
411 jpec_enc_block_quant(e);
412 jpec_enc_block_zz(e);
413 e->hskel->encode_block(e->hskel->opq, &e->block, e->buf);
414 }
415 jpec_enc_close(e);
416 *len = e->buf->len;
417 return e->buf->stream;
418}

420/* Update the internal quantization matrix according to the asked quality */
421static void jpec_enc_init_dqt(jpec_enc_t* e) {
422 assert(e);
423 float qualf = (float)e->qual;
424 float scale = (e->qual < 50) ? (50 / qualf) : (2 - qualf / 50);
425 for (int i = 0; i < 64; i++) {
426 int a = (int)((float)jpec_qzr[i] * scale + 0.5);
427 a = (a < 1) ? 1 : ((a > 255) ? 255 : a);
428 e->dqt[i] = a;
429 }
430}

472static void jpec_enc_write_dqt(jpec_enc_t* e) {
473 assert(e);

530static int jpec_enc_next_block(jpec_enc_t* e) {
531 assert(e);
532 int rv = (++e->bnum >= e->bmax) ? 0 : 1;
533 if (rv) {
534 e->bx = (e->bnum << 3) % e->w8;
535 e->by = ((e->bnum << 3) / e->w8) << 3;
536 }
537 return rv;
538}

540static void jpec_enc_block_dct(jpec_enc_t* e) {
541 assert(e && e->bnum >= 0);
542#define JPEC_BLOCK(col,row) e->img[(((e->by + row) < e->h) ? e->by + row : e->h-1) * \
543 e->w + (((e->bx + col) < e->w) ? e->bx + col : e->w-1)]
544 const float* coeff = jpec_dct;
545 float tmp[64];
546 for (int row = 0; row < 8; row++) {
547 /* NOTE: the shift by 256 allows resampling from [0 255] to [-128 127] */
548 float s0 = (float)(JPEC_BLOCK(0, row) + JPEC_BLOCK(7, row) - 256);
549 float s1 = (float)(JPEC_BLOCK(1, row) + JPEC_BLOCK(6, row) - 256);
550 float s2 = (float)(JPEC_BLOCK(2, row) + JPEC_BLOCK(5, row) - 256);
551 float s3 = (float)(JPEC_BLOCK(3, row) + JPEC_BLOCK(4, row) - 256);
552
553 float d0 = (float)(JPEC_BLOCK(0, row) - JPEC_BLOCK(7, row));
554 float d1 = (float)(JPEC_BLOCK(1, row) - JPEC_BLOCK(6, row));
555 float d2 = (float)(JPEC_BLOCK(2, row) - JPEC_BLOCK(5, row));
556 float d3 = (float)(JPEC_BLOCK(3, row) - JPEC_BLOCK(4, row));
557
558 tmp[8 * row] = coeff[3] * (s0 + s1 + s2 + s3);
559 tmp[8 * row + 1] = coeff[0] * d0 + coeff[2] * d1 + coeff[4] * d2 + coeff[6] * d3;
560 tmp[8 * row + 2] = coeff[1] * (s0 - s3) + coeff[5] * (s1 - s2);
561 tmp[8 * row + 3] = coeff[2] * d0 - coeff[6] * d1 - coeff[0] * d2 - coeff[4] * d3;
562 tmp[8 * row + 4] = coeff[3] * (s0 - s1 - s2 + s3);
563 tmp[8 * row + 5] = coeff[4] * d0 - coeff[0] * d1 + coeff[6] * d2 + coeff[2] * d3;
564 tmp[8 * row + 6] = coeff[5] * (s0 - s3) + coeff[1] * (s2 - s1);
565 tmp[8 * row + 7] = coeff[6] * d0 - coeff[4] * d1 + coeff[2] * d2 - coeff[0] * d3;
566 }
567 for (int col = 0; col < 8; col++) {
568 float s0 = tmp[col] + tmp[56 + col];
569 float s1 = tmp[8 + col] + tmp[48 + col];
570 float s2 = tmp[16 + col] + tmp[40 + col];
571 float s3 = tmp[24 + col] + tmp[32 + col];
572
573 float d0 = tmp[col] - tmp[56 + col];
574 float d1 = tmp[8 + col] - tmp[48 + col];
575 float d2 = tmp[16 + col] - tmp[40 + col];
576 float d3 = tmp[24 + col] - tmp[32 + col];
577
578 e->block.dct[col] = coeff[3] * (s0 + s1 + s2 + s3);
579 e->block.dct[8 + col] = coeff[0] * d0 + coeff[2] * d1 + coeff[4] * d2 + coeff[6] * d3;
580 e->block.dct[16 + col] = coeff[1] * (s0 - s3) + coeff[5] * (s1 - s2);
581 e->block.dct[24 + col] = coeff[2] * d0 - coeff[6] * d1 - coeff[0] * d2 - coeff[4] * d3;
582 e->block.dct[32 + col] = coeff[3] * (s0 - s1 - s2 + s3);
583 e->block.dct[40 + col] = coeff[4] * d0 - coeff[0] * d1 + coeff[6] * d2 + coeff[2] * d3;
584 e->block.dct[48 + col] = coeff[5] * (s0 - s3) + coeff[1] * (s2 - s1);
585 e->block.dct[56 + col] = coeff[6] * d0 - coeff[4] * d1 + coeff[2] * d2 - coeff[0] * d3;
586 }
587#undef JPEC_BLOCK
588}
