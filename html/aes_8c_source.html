28size_t aes256_cbc_encrypt(const unsigned char aes_key[32], const unsigned char iv[AES_BLOCK_SIZE], const unsigned char* data, size_t size, int pad, unsigned char* out)
29{
30 size_t written = 0;
31 int padsize = size % AES_BLOCK_SIZE;
32 unsigned char mixed[AES_BLOCK_SIZE];
33
34 if (!data || !size || !out)
35 return 0;
36
37 if (!pad && padsize != 0)
38 return 0;
39
40 AES256_ctx aes_ctx;
41 // Set cipher key
42 AES256_init(&aes_ctx, aes_key);
43
44 memcpy_safe(mixed, iv, AES_BLOCK_SIZE);
45
46 // Write all but the last block
47 while (written + AES_BLOCK_SIZE <= size) {
48 for (int i = 0; i != AES_BLOCK_SIZE; i++)
49 mixed[i] ^= *data++;
50 AES256_encrypt(&aes_ctx, 1, out + written, mixed);
51 memcpy_safe(mixed, out + written, AES_BLOCK_SIZE);
52 written += AES_BLOCK_SIZE;
53 }
54 if (pad) {
55 // For all that remains, pad each byte with the value of the remaining
56 // space. If there is none, pad by a full block.
57 for (int i = 0; i != padsize; i++)
58 mixed[i] ^= *data++;
59 for (int i = padsize; i != AES_BLOCK_SIZE; i++)
60 mixed[i] ^= AES_BLOCK_SIZE - padsize;
61 AES256_encrypt(&aes_ctx, 1, out + written, mixed);
62 written += AES_BLOCK_SIZE;
63 }
64 return written;
65}

67size_t aes256_cbc_decrypt(const unsigned char aes_key[32], const unsigned char iv[AES_BLOCK_SIZE], const unsigned char* data, size_t size, int pad, unsigned char* out)
68{
69 unsigned char padsize = 0;
70 size_t written = 0;
71 int fail = 0;
72 const unsigned char* prev = iv;
73
74 if (!data || !size || !out)
75 return 0;
76
77 if (size % AES_BLOCK_SIZE != 0)
78 return 0;
79
80 AES256_ctx aes_ctx;
81 // Set cipher key
82 AES256_init(&aes_ctx, aes_key);
83
84 // Decrypt all data. Padding will be checked in the output.
85 while (written != size) {
86 AES256_decrypt(&aes_ctx, 1, out, data + written);
87 for (int i = 0; i != AES_BLOCK_SIZE; i++)
88 *out++ ^= prev[i];
89 prev = data + written;
90 written += AES_BLOCK_SIZE;
91 }
92
93 // When decrypting padding, attempt to run in constant-time
94 if (pad) {
95 // If used, padding size is the value of the last decrypted byte. For
96 // it to be valid, It must be between 1 and AES_BLOCK_SIZE.
97 padsize = *--out;
98 fail = !padsize | (padsize > AES_BLOCK_SIZE);
99
100 // If not well-formed, treat it as though there's no padding.
101 padsize *= !fail;
102
103 // All padding must equal the last byte otherwise it's not well-formed
104 for (int i = AES_BLOCK_SIZE; i != 0; i--)
105 fail |= ((i > AES_BLOCK_SIZE - padsize) & (*out-- != padsize));
106
107 written -= padsize;
108 }
109 return written * !fail;
110}
