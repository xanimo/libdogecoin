79int dogecoin_block_header_deserialize(dogecoin_block_header* header, struct const_buffer* buf) {
80 if (!deser_s32(&header->version, buf))
81 return false;
82 if (!deser_u256(header->prev_block, buf))
83 return false;
84 if (!deser_u256(header->merkle_root, buf))
85 return false;
86 if (!deser_u32(&header->timestamp, buf))
87 return false;
88 if (!deser_u32(&header->bits, buf))
89 return false;
90 if (!deser_u32(&header->nonce, buf))
91 return false;
92 return true;
93}

140dogecoin_bool dogecoin_block_header_hash(dogecoin_block_header* header, uint256 hash) {
141 cstring* s = cstr_new_sz(80);
142 dogecoin_block_header_serialize(s, header);
143 sha256_raw((const uint8_t*)s->str, s->len, hash);
144 sha256_raw(hash, SHA256_DIGEST_LENGTH, hash);
145 cstr_free(s, true);
146 dogecoin_bool ret = true;
147 return ret;
