109void utils_hex_to_bin(const char* str, unsigned char* out, size_t inLen, size_t* outLen)
110 {
111 size_t bLen = inLen / 2;
112 size_t i;
113 dogecoin_mem_zero(out, bLen);
114 for (i = 0; i < bLen; i++) {
115 if (str[i * 2] >= '0' && str[i * 2] <= '9') {
116 *out = (str[i * 2] - '0') << 4;
117 }
118 if (str[i * 2] >= 'a' && str[i * 2] <= 'f') {
119 *out = (10 + str[i * 2] - 'a') << 4;
120 }
121 if (str[i * 2] >= 'A' && str[i * 2] <= 'F') {
122 *out = (10 + str[i * 2] - 'A') << 4;
123 }
124 if (str[i * 2 + 1] >= '0' && str[i * 2 + 1] <= '9') {
125 *out |= (str[i * 2 + 1] - '0');
126 }
127 if (str[i * 2 + 1] >= 'a' && str[i * 2 + 1] <= 'f') {
128 *out |= (10 + str[i * 2 + 1] - 'a');
129 }
130 if (str[i * 2 + 1] >= 'A' && str[i * 2 + 1] <= 'F') {
131 *out |= (10 + str[i * 2 + 1] - 'A');
132 }
133 out++;
134 }
135 *outLen = i;
136 }

147uint8_t* utils_hex_to_uint8(const char* str)
148 {
149 uint8_t c;
150 size_t i;
151 if (strlens(str) > TO_UINT8_HEX_BUF_LEN) {
152 return NULL;
153 }
154 dogecoin_mem_zero(buffer_hex_to_uint8, TO_UINT8_HEX_BUF_LEN);
155 for (i = 0; i < strlens(str) / 2; i++) {
156 c = 0;
157 if (str[i * 2] >= '0' && str[i * 2] <= '9') {
158 c += (str[i * 2] - '0') << 4;
159 }
160 if (str[i * 2] >= 'a' && str[i * 2] <= 'f') {
161 c += (10 + str[i * 2] - 'a') << 4;
162 }
163 if (str[i * 2] >= 'A' && str[i * 2] <= 'F') {
164 c += (10 + str[i * 2] - 'A') << 4;
165 }
166 if (str[i * 2 + 1] >= '0' && str[i * 2 + 1] <= '9') {
167 c += (str[i * 2 + 1] - '0');
168 }
169 if (str[i * 2 + 1] >= 'a' && str[i * 2 + 1] <= 'f') {
170 c += (10 + str[i * 2 + 1] - 'a');
171 }
172 if (str[i * 2 + 1] >= 'A' && str[i * 2 + 1] <= 'F') {

191void utils_bin_to_hex(unsigned char* bin_in, size_t inlen, char* hex_out)
192 {
193 static char digits[] = "0123456789abcdef";

296void utils_uint256_sethex(char* psz, uint8_t* out)
297{
298 dogecoin_mem_zero(out, sizeof(uint256));
299
300 // skip leading spaces
301 while (isspace(*psz)) {
302 psz++;
303 }
304
305 // skip 0x
306 if (psz[0] == '0' && tolower(psz[1]) == 'x') {
307 psz += 2;
308 }
309
310 // hex string to uint
311 const char* pbegin = psz;
312 while (utils_hex_digit(*psz) != -1) {
313 psz++;
314 }
315 psz--;
316 unsigned char* p1 = (unsigned char*)out;
317 unsigned char* pend = p1 + sizeof(uint256);
318 while (psz >= pbegin && p1 < pend) {
319 *p1 = utils_hex_digit(*psz--);
320 if (psz >= pbegin) {
321 *p1 |= ((unsigned char)utils_hex_digit(*psz--) << 4);
322 p1++;
323 }
324 }
325 }

549/* itoa: convert n to characters in s */
550void dogecoin_uitoa(int n, char s[])
551{
552 int i, sign;
553
554 if ((sign = n) < 0) /* record sign */
555 n = -n; /* make n positive */
556 i = 0;
557 do { /* generate digits in reverse order */
558 s[i++] = n % 10 + '0'; /* get next digit */
559 } while ((n /= 10) > 0); /* delete it */
560 if (sign < 0)
561 s[i++] = '-';
562 s[i] = '\0';
563 dogecoin_str_reverse(s);
564}

576unsigned int base64_int(unsigned int ch) {
577
578 // ASCII to base64_int
579 // 65-90 Upper Case >> 0-25
580 // 97-122 Lower Case >> 26-51
581 // 48-57 Numbers >> 52-61
582 // 43 Plus (+) >> 62
583 // 47 Slash (/) >> 63
584 // 61 Equal (=) >> 64~
585 if (ch==43)
586 return 62;
587 if (ch==47)
588 return 63;
589 if (ch==61)
590 return 64;
591 if ((ch>47) && (ch<58))
592 return ch + 4;
593 if ((ch>64) && (ch<91))
594 return ch - 'A';
595 if ((ch>96) && (ch<123))
596 return (ch - 'a') + 26;
597 return 0;
598}

616unsigned int base64_encode(const unsigned char* in, unsigned int in_len, unsigned char* out) {
617
618 unsigned int i=0, j=0, k=0, s[3];
619
620 for (i=0;i<in_len;i++) {
621 s[j++]=*(in+i);
622 if (j==3) {
623 out[k+0] = base64_char[ (s[0]&255)>>2 ];
624 out[k+1] = base64_char[ ((s[0]&0x03)<<4)+((s[1]&0xF0)>>4) ];
625 out[k+2] = base64_char[ ((s[1]&0x0F)<<2)+((s[2]&0xC0)>>6) ];
626 out[k+3] = base64_char[ s[2]&0x3F ];
627 j=0; k+=4;
628 }
629 }
630
631 if (j) {
632 if (j==1)
633 s[1] = 0;
634 out[k+0] = base64_char[ (s[0]&255)>>2 ];
635 out[k+1] = base64_char[ ((s[0]&0x03)<<4)+((s[1]&0xF0)>>4) ];
636 if (j==2)
637 out[k+2] = base64_char[ ((s[1]&0x0F)<<2) ];
638 else
639 out[k+2] = '=';
640 out[k+3] = '=';
641 k+=4;
642 }
643
644 out[k] = '\0';
645
646 return k;
647}

649unsigned int base64_decode(const unsigned char* in, unsigned int in_len, unsigned char* out) {
650
651 unsigned int i=0, j=0, k=0, s[4];
652
653 for (i=0;i<in_len;i++) {
654 s[j++]=base64_int(*(in+i));
655 if (j==4) {
656 out[k+0] = ((s[0]&255)<<2)+((s[1]&0x30)>>4);
657 if (s[2]!=64) {
658 out[k+1] = ((s[1]&0x0F)<<4)+((s[2]&0x3C)>>2);
659 if ((s[3]!=64)) {
660 out[k+2] = ((s[2]&0x03)<<6)+(s[3]); k+=3;
661 } else {
662 k+=2;
663 }
664 } else {
665 k+=1;
666 }
667 j=0;
668 }
669 }
670
671 out[k] = '\0';
672
673 return k;
