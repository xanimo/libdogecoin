209void* memcpy_safe(void* destination, const void* source, size_t count) {
210 char *pszDest = (char *)destination;
211 const char *pszSource =( const char*)source;
212 if((pszDest!= NULL) && (pszSource!= NULL)) {
213 while(count) {
214 *(pszDest++)= *(pszSource++);
215 --count;
216 }
217 }
218 return destination;
219}

221errno_t memset_safe(volatile void *v, rsize_t smax, int c, rsize_t n) {
222 if (v == NULL) return EINVAL;
223 if (smax > RSIZE_MAX) return EINVAL;
224 if (n > smax) return EINVAL;
225
226 volatile unsigned char *p = v;
227 while (smax-- && n--) {
228 *p++ = c;
229 }
230
231 return 0;
232}
