57dogecoin_bool addresses_from_pubkey(const dogecoin_chainparams* chain, const char* pubkey_hex, char* p2pkh_address)
58{
59 if (!pubkey_hex)
60 return false;
61 if (strlen(pubkey_hex) != 66)
62 return false;
63 dogecoin_pubkey pubkey;
64 dogecoin_pubkey_init(&pubkey);
65 pubkey.compressed = 1;
66 size_t outlen = 0;
67 utils_hex_to_bin(pubkey_hex, pubkey.pubkey, strlen(pubkey_hex), &outlen);
68 assert(dogecoin_pubkey_is_valid(&pubkey) == 1);
69 dogecoin_pubkey_getaddr_p2pkh(&pubkey, chain, p2pkh_address);
70 return true;
71}

83dogecoin_bool pubkey_from_privatekey(const dogecoin_chainparams* chain, const char* privkey_wif, char* pubkey_hex, size_t* sizeout)
84{
85 dogecoin_key key;
86 dogecoin_privkey_init(&key);
87 if (!dogecoin_privkey_decode_wif(privkey_wif, chain, &key))
88 return false;
89 dogecoin_pubkey pubkey;
90 dogecoin_pubkey_init(&pubkey);
91 assert(dogecoin_pubkey_is_valid(&pubkey) == 0);
92 dogecoin_pubkey_from_key(&key, &pubkey);
93 assert(dogecoin_pubkey_is_valid(&pubkey) == 1);
94 dogecoin_privkey_cleanse(&key);
95 dogecoin_pubkey_get_hex(&pubkey, pubkey_hex, sizeout);
96 dogecoin_pubkey_cleanse(&pubkey);
97 return true;
98}

111dogecoin_bool gen_privatekey(const dogecoin_chainparams* chain, char* privkey_wif, size_t strsize_wif, char* privkey_hex_or_null)
112{
113 uint8_t pkeybase58c[34];
114 pkeybase58c[0] = chain->b58prefix_secret_address;
115 pkeybase58c[33] = 1; /* always use compressed keys */
116 dogecoin_key key;
117 dogecoin_privkey_init(&key);
118 dogecoin_privkey_gen(&key);
119 memcpy_safe(&pkeybase58c[1], key.privkey, DOGECOIN_ECKEY_PKEY_LENGTH);
120 assert(dogecoin_base58_encode_check(pkeybase58c, 34, privkey_wif, strsize_wif) != 0);
121 // also export the hex privkey if use had passed in a valid pointer
122 // will always export 32 bytes
123 if (privkey_hex_or_null != NULL)
124 utils_bin_to_hex(key.privkey, DOGECOIN_ECKEY_PKEY_LENGTH, privkey_hex_or_null);
125 dogecoin_privkey_cleanse(&key);
126 return true;
127}

138dogecoin_bool hd_gen_master(const dogecoin_chainparams* chain, char* masterkeyhex, size_t strsize)
139{
140 dogecoin_hdnode node;
141 uint8_t seed[32];
142 dogecoin_random_bytes(seed, 32, true);
143 dogecoin_hdnode_from_seed(seed, 32, &node);
144 dogecoin_mem_zero(seed, 32);
145 dogecoin_hdnode_serialize_private(&node, chain, masterkeyhex, strsize);
146 dogecoin_mem_zero(&node, sizeof(node));
147 return true;
148}

158dogecoin_bool hd_print_node(const dogecoin_chainparams* chain, const char* nodeser)
159{
160 dogecoin_hdnode node;
161 if (!dogecoin_hdnode_deserialize(nodeser, chain, &node))
162 return false;
163
164 char str[128];
165 size_t strsize = sizeof(str);
166 printf("ext key: %s\n", nodeser);
167
168 uint8_t pkeybase58c[34];
169 dogecoin_hdnode_serialize_public(&node, chain, str, strsize);
170 printf("extended pubkey: %s\n", str);
171
172 if (!dogecoin_hdnode_get_pub_hex(&node, str, &strsize))
173 return false;
174 printf("pubkey hex: %s\n", str);
175
176 pkeybase58c[0] = chain->b58prefix_secret_address;
177 pkeybase58c[33] = 1; /* always use compressed keys */
178 char privkey_wif[128];
179 memcpy_safe(&pkeybase58c[1], node.private_key, DOGECOIN_ECKEY_PKEY_LENGTH);
180 assert(dogecoin_base58_encode_check(pkeybase58c, sizeof(pkeybase58c), privkey_wif, sizeof(privkey_wif)) != 0);
181 if (dogecoin_hdnode_has_privkey(&node)) {
182 printf("privatekey WIF: %s\n", privkey_wif);
183 }
184 printf("depth: %d\n", node.depth);
185 printf("child index: %d\n", node.child_num);
186 char addr[34 + 1];
187 addresses_from_pubkey(&dogecoin_chainparams_main, str, addr);
188 printf("p2pkh address: %s\n", addr);
189 return true;
190}

204dogecoin_bool hd_derive(const dogecoin_chainparams* chain, const char* masterkey, const char* derived_path, char* extkeyout, size_t extkeyout_size)
205{
206 if (!derived_path || !masterkey || !extkeyout)
207 return false;
208 dogecoin_hdnode node, nodenew;
209 if (!dogecoin_hdnode_deserialize(masterkey, chain, &node))
210 return false;
211 //check if we only have the publickey
212 bool pubckd = !dogecoin_hdnode_has_privkey(&node);
213 //derive child key, use pubckd or privckd
214 if (!dogecoin_hd_generate_key(&nodenew, derived_path, pubckd ? node.public_key : node.private_key, node.chain_code, pubckd))
215 return false;
216 if (pubckd)
217 dogecoin_hdnode_serialize_public(&nodenew, chain, extkeyout, extkeyout_size);
218 else
219 dogecoin_hdnode_serialize_private(&nodenew, chain, extkeyout, extkeyout_size);
220 return true;
221}
