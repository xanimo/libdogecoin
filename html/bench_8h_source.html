19static int64_t gettime_i64(void) {
20 struct timeval tv;
21 gettimeofday(&tv, NULL);
22 return (int64_t)tv.tv_usec + (int64_t)tv.tv_sec * 1000000LL;

28/* Format fixed point number. */
29void print_number(const int64_t x) {
30 int64_t x_abs, y;
31 int c, i, rounding, g; /* g = integer part size, c = fractional part size */
32 size_t ptr;
33 char buffer[30];
34
35 if (x == INT64_MIN) {
36 /* Prevent UB. */
37 printf("ERR");
38 return;
39 }
40 x_abs = x < 0 ? -x : x;
41
42 /* Determine how many decimals we want to show (more than FP_EXP makes no
43 * sense). */
44 y = x_abs;
45 c = 0;
46 while (y > 0LL && y < 100LL * FP_MULT && c < FP_EXP) {
47 y *= 10LL;
48 c++;
49 }
50
51 /* Round to 'c' decimals. */
52 y = x_abs;
53 rounding = 0;
54 for (i = c; i < FP_EXP; ++i) {
55 rounding = (y % 10) >= 5;
56 y /= 10;
57 }
58 y += rounding;
59
60 /* Format and print the number. */
61 ptr = sizeof(buffer) - 1;
62 buffer[ptr] = 0;
63 g = 0;
64 if (c != 0) { /* non zero fractional part */
65 for (i = 0; i < c; ++i) {
66 buffer[--ptr] = '0' + (y % 10);
67 y /= 10;
68 }
69 } else if (c == 0) { /* fractional part is 0 */
70 buffer[--ptr] = '0';
71 }
72 buffer[--ptr] = '.';
73 do {
74 buffer[--ptr] = '0' + (y % 10);
75 y /= 10;
76 g++;
77 } while (y != 0);
78 if (x < 0) {
79 buffer[--ptr] = '-';
80 g++;
81 }
82 printf("%5.*s", g, &buffer[ptr]); /* Prints integer part */
83 printf("%-*s", FP_EXP, &buffer[ptr + g]); /* Prints fractional part */
84}

86void run_benchmark(char *name, void (*benchmark)(void*, int), void (*setup)(void*), void (*teardown)(void*, int), void* data, int count, int iter) {
87 int i;
88 int64_t min = INT64_MAX;
89 int64_t sum = 0;
90 int64_t max = 0;
91 for (i = 0; i < count; i++) {
92 int64_t begin, total;
93 if (setup != NULL) {
94 setup(data);
95 }
96 begin = gettime_i64();
97 benchmark(data, iter);
98 total = gettime_i64() - begin;
99 if (teardown != NULL) {
100 teardown(data, iter);
101 }
102 if (total < min) {
103 min = total;
104 }
105 if (total > max) {
106 max = total;
107 }
108 sum += total;
109 }
110 /* ',' is used as a column delimiter */
111 printf("%-30s, ", name);
112 print_number(min * FP_MULT / iter);
113 printf(" , ");
114 print_number(((sum * FP_MULT) / count) / iter);
115 printf(" , ");
116 print_number(max * FP_MULT / iter);
117 printf("\n");
118}

132/* takes an array containing the arguments that the user is allowed to enter on the command-line
133 returns:
134 - 1 if the user entered an invalid argument
135 - 0 if all the user entered arguments are valid */
136int have_invalid_args(int argc, char** argv, char** valid_args, size_t n) {
137 size_t i;
138 int found_valid;
139 char** argm = argv + argc;
140 argv++;
141
142 while (argv != argm) {
143 found_valid = 0;
144 for (i = 0; i < n; i++) {
145 if (strcmp(*argv, valid_args[i]) == 0) {
146 found_valid = 1; /* user entered a valid arg from the list */
147 break;
148 }
149 }
150 if (found_valid == 0) {
151 return 1; /* invalid arg found */
152 }
153 argv++;
154 }
155 return 0;
156}
