45/* IPv4 addresses are mapped to 16bytes with a prefix of 10 x 0x00 + 2 x 0xff */
46static const uint8_t DOGECOIN_IPV4_PREFIX[12] = {0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0xff, 0xff};

55static inline dogecoin_bool is_ipv4_mapped(const unsigned char* ipaddr)
56{
57 return memcmp(ipaddr, DOGECOIN_IPV4_PREFIX, 12) == 0;
58}

80cstring* dogecoin_p2p_message_new(const unsigned char netmagic[4], const char* command, const void* data, uint32_t data_len)
81{
82 cstring* s = cstr_new_sz(DOGECOIN_P2P_HDRSZ + data_len);
83
84 /* network identifier (magic number) */
85 cstr_append_buf(s, netmagic, 4);
86
87 /* command string */
88 char command_null[12];
89 dogecoin_mem_zero(command_null, 12);
90 memcpy_safe(command_null, command, strlen(command));
91 //memset(command_null+strlen(command), 0, 12-strlen(command));
92 cstr_append_buf(s, command_null, 12);
93
94 /* data length, always 4 bytes */
95 uint32_t data_len_le = htole32(data_len);
96 cstr_append_buf(s, &data_len_le, 4);
97
98 /* data checksum (first 4 bytes of the double sha256 hash of the pl) */
99 uint256 msghash;
100 dogecoin_hash(data, data_len, msghash);
101 cstr_append_buf(s, &msghash[0], 4);
102
103 /* data payload */
104 if (data_len > 0)
105 cstr_append_buf(s, data, data_len);
106
107 return s;
108}

119dogecoin_bool dogecoin_p2p_deser_addr(unsigned int protocol_version, dogecoin_p2p_address* addr, struct const_buffer* buf)
120{
121 if (protocol_version >= DOGECOIN_ADDR_TIME_VERSION) {
122 if (!deser_u32(&addr->time, buf))
123 return false;
124 } else
125 addr->time = 0;
126
127 if (!deser_u64(&addr->services, buf))
128 return false;
129 if (!deser_bytes(&addr->ip, buf, 16))
130 return false;
131 if (!deser_u16(&addr->port, buf))
132 return false;
133 return true;
134}

159void dogecoin_addr_to_p2paddr(struct sockaddr* addr, dogecoin_p2p_address* addr_out)
160{
161 if (addr->sa_family == AF_INET6) {
162 struct sockaddr_in6* saddr = (struct sockaddr_in6*)addr;
163 memcpy_safe(&addr_out->ip, &saddr->sin6_addr, 16);
164 addr_out->port = ntohs(saddr->sin6_port);
165 } else if (addr->sa_family == AF_INET) {
166 struct sockaddr_in* saddr = (struct sockaddr_in*)addr;
167 dogecoin_mem_zero(&addr_out->ip[0], 10);
168 memset(&addr_out->ip[10], 0xff, 2);
169 memcpy_safe(&addr_out->ip[12], &saddr->sin_addr, 4);
170 addr_out->port = ntohs(saddr->sin_port);
171 }
172}

182void dogecoin_p2paddr_to_addr(dogecoin_p2p_address* p2p_addr, struct sockaddr* addr_out)
183{
184 if (!p2p_addr || !addr_out)
185 return;
186
187 if (!is_ipv4_mapped(p2p_addr->ip)) {
188 /* ipv6 */
189 struct sockaddr_in6* saddr = (struct sockaddr_in6*)addr_out;
190 memcpy_safe(&saddr->sin6_addr, p2p_addr->ip, 16);
191 saddr->sin6_port = htons(p2p_addr->port);
192 } else {
193 struct sockaddr_in* saddr = (struct sockaddr_in*)addr_out;
194 memcpy_safe(&saddr->sin_addr, &p2p_addr->ip[12], 4);
195 saddr->sin_port = htons(p2p_addr->port);
196 }
197}

208void dogecoin_p2p_msg_version_init(dogecoin_p2p_version_msg* msg, const dogecoin_p2p_address* addrFrom, const dogecoin_p2p_address* addrTo, const char* strSubVer, dogecoin_bool relay)
209{
210 msg->version = DOGECOIN_PROTOCOL_VERSION;
211 msg->services = 0;
212 msg->timestamp = time(NULL);
213 msg->timestamp = time(NULL);
214 if (addrTo)
215 msg->addr_recv = *addrTo;
216 else
217 dogecoin_p2p_address_init(&msg->addr_recv);
218 if (addrFrom)
219 msg->addr_from = *addrFrom;
220 else
221 dogecoin_p2p_address_init(&msg->addr_from);
222 dogecoin_cheap_random_bytes((uint8_t*)&msg->nonce, sizeof(msg->nonce));
223 if (strSubVer && strlen(strSubVer) < 128)
224 memcpy_safe(msg->useragent, strSubVer, strlen(strSubVer));
225
226 msg->start_height = 0;
227 msg->relay = relay;
228}

257dogecoin_bool dogecoin_p2p_msg_version_deser(dogecoin_p2p_version_msg* msg, struct const_buffer* buf)
258{
259 dogecoin_mem_zero(msg, sizeof(*msg));
260 if (!deser_s32(&msg->version, buf))
261 return false;
262 if (!deser_u64(&msg->services, buf))
263 return false;
264 if (!deser_s64(&msg->timestamp, buf))
265 return false;
266 if (!dogecoin_p2p_deser_addr(0, &msg->addr_recv, buf))
267 return false;
268 if (!dogecoin_p2p_deser_addr(0, &msg->addr_from, buf))
269 return false;
270 if (!deser_u64(&msg->nonce, buf))
271 return false;
272
273 uint32_t ua_len;
274 uint32_t cpy_len;
275 if (!deser_varlen(&ua_len, buf))
276 return false;
277
278 cpy_len = ua_len;
279 if (cpy_len > 128)
280 cpy_len = 128;
281 /* we only support user agent strings up to 1024 chars */
282 /* we only copy 128bytes to the message structure */
283 if (ua_len > 1024)
284 return false;
285
286 char ua_str[1024];
287 if (!deser_bytes(ua_str, buf, ua_len))
288 return false;
289 dogecoin_mem_zero(msg->useragent, sizeof(msg->useragent));
290 memcpy_safe(&msg->useragent, ua_str, cpy_len);
291
292 if (!deser_s32(&msg->start_height, buf))
293 return false;
294 if (msg->version > 70003)
295 if (!deser_bytes(&msg->relay, buf, 1))
296 return false;
297
298 return true;
299}

351void dogecoin_p2p_msg_getheaders(vector* blocklocators, uint256 hashstop, cstring* s)
352{
353 unsigned int i;
354
355 ser_u32(s, DOGECOIN_PROTOCOL_VERSION);
356 ser_varlen(s, blocklocators->len);
357 for (i = 0; i < blocklocators->len; i++) {
358 uint256 *hash = vector_idx(blocklocators, i);
359 ser_bytes(s, hash, DOGECOIN_HASH_LENGTH);
360 }
361 if (hashstop)
362 ser_bytes(s, hashstop, DOGECOIN_HASH_LENGTH);
363 else
364 ser_bytes(s, NULLHASH, DOGECOIN_HASH_LENGTH);
365}

376dogecoin_bool dogecoin_p2p_deser_msg_getheaders(vector* blocklocators, uint256 hashstop, struct const_buffer* buf)
377{
378 int32_t version;
379 uint32_t vsize;
380 if (!deser_s32(&version, buf))
381 return false;
382 if (!deser_varlen(&vsize, buf))
383 return false;
384 vector_resize(blocklocators, vsize);
385 for (unsigned int i = 0; i < vsize; i++) {
386 uint256 *hash = dogecoin_malloc(DOGECOIN_HASH_LENGTH);
387 if (!deser_u256(*hash, buf)) {
388 dogecoin_free(hash);
389 return false;
390 }
391 vector_add(blocklocators, hash);
392 }
393 if (!deser_u256(hashstop, buf))
394 return false;
395 return true;
396}
