41void hash_message(char* msg, uint256 message_bytes) {
42 size_t msg_magic_len = strlen(msg_magic), msg_len = strlen(msg);
43 char* tmp_msg = dogecoin_char_vla(msg_magic_len + msg_len + 7);
44 tmp_msg[0] = msg_magic_len;
45 memcpy_safe(tmp_msg + 1, msg_magic, msg_magic_len);
46 tmp_msg[msg_magic_len + 1] = msg_len;
47 memcpy_safe(tmp_msg + msg_magic_len + 2, msg, msg_len + 1);
48 dogecoin_hash((const unsigned char*)tmp_msg, strlen(tmp_msg), message_bytes);
49 dogecoin_free(tmp_msg);
50}

61char* sign_message(char* privkey, char* msg) {
62 if (!privkey || !msg) return false;
63
64 uint256 message_bytes;
65 hash_message(msg, message_bytes);
66
67 size_t compact_signature_length = 65;
68 unsigned char* compact_signature = dogecoin_uchar_vla(compact_signature_length);
69
70 dogecoin_key key;
71 dogecoin_pubkey pubkey;
72 if (!init_keypair(privkey, &key, &pubkey)) return false;
73
74 int recid = -1;
75
76 if (!dogecoin_key_sign_hash_compact_recoverable_fcomp(&key, message_bytes, compact_signature, &compact_signature_length, &recid)) return false;
77 if (!dogecoin_key_recover_pubkey((const unsigned char*)compact_signature, message_bytes, recid, &pubkey)) return false;
78
79 char p2pkh_address[35];
80 if (!dogecoin_pubkey_getaddr_p2pkh(&pubkey, &dogecoin_chainparams_main, p2pkh_address)) return false;
81
82 unsigned char* base64_encoded_output = dogecoin_uchar_vla(1+(sizeof(char)*base64_encoded_size(compact_signature_length)));
83 base64_encode((unsigned char*)compact_signature, compact_signature_length, base64_encoded_output);
84
85 dogecoin_free(compact_signature);
86 dogecoin_privkey_cleanse(&key);
87 dogecoin_pubkey_cleanse(&pubkey);
88
89 return (char*)base64_encoded_output;
90}

102int verify_message(char* sig, char* msg, char* address) {
103 if (!(sig || msg || address)) return false;
104
105 uint256 message_bytes;
106 hash_message(msg, message_bytes);
107
108 size_t encoded_length = strlen((const char*)sig);
109 unsigned char* decoded_signature = dogecoin_uchar_vla(base64_decoded_size(encoded_length+1)+1);
110 base64_decode((unsigned char*)sig, encoded_length, decoded_signature);
111
112 dogecoin_pubkey pub_key;
113 dogecoin_pubkey_init(&pub_key);
114 pub_key.compressed = false;
115
116 int header = decoded_signature[0] & 0xFF;
117
118 if (header < 27 || header > 42) return false;
119
120 if (header >= 31) {
121 pub_key.compressed = true;
122 header -= 4;
123 }
124
125 int recid = header - 27;
126 if (!dogecoin_key_recover_pubkey((const unsigned char*)decoded_signature, message_bytes, recid, &pub_key)) return false;
127 if (!dogecoin_pubkey_verify_sigcmp(&pub_key, message_bytes, decoded_signature)) return false;
128
129 char p2pkh_address[35];
130 const dogecoin_chainparams* chain = chain_from_b58_prefix(address);
131 if (!dogecoin_pubkey_getaddr_p2pkh(&pub_key, chain, p2pkh_address)) return false;
132
133 dogecoin_free(decoded_signature);
134 dogecoin_pubkey_cleanse(&pub_key);
135 return strcmp(p2pkh_address, address)==0;
136}
